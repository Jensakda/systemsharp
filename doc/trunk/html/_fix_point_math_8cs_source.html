201  public override bool Rewrite(CodeDescriptor decompilee, MethodBase callee, StackElement[] args, IDecompiler stack, IFunctionBuilder builder)
202  {
203  if (args[0].Sample == null)
204  return false;
205 
206  ISized one, rsample;
207  if (IsSigned)
208  {
209  Signed sample = (Signed)args[0].Sample;
210  Signed sone = Signed.FromLong(1, (int)sample.Size);
211  one = sone;
212  rsample = sample + sone;
213  }
214  else
215  {
216  Unsigned sample = (Unsigned)args[0].Sample;
217  Unsigned uone = Unsigned.FromULong(1, (int)sample.Size);
218  one = uone;
219  rsample = sample + uone;
220  }
221  LiteralReference oneLit = LiteralReference.CreateConstant(one);
222  Expression inc;
223  if (IsDecrement)
224  inc = args[0].Expr - oneLit;
225  else
226  inc = args[0].Expr + oneLit;
227  inc.ResultType = TypeDescriptor.GetTypeOf(rsample);
228  inc = IntrinsicFunctions.Resize(inc, (int)one.Size, TypeDescriptor.GetTypeOf(one));
229  stack.Push(new StackElement(inc, one, Analysis.Msil.EVariability.ExternVariable));
230  return true;
