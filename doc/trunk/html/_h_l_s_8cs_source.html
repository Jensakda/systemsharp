662  private static void ComputeCStepsForBranchTargets(XILSchedulingAdapter xsa)
663  {
664  var cfg = xsa.CFG;

666  {
667  XIL3Instr xil3i = cfg.Instructions[i];

742 
743  var clk = proc.Sensitivity[0];
744  SignalBase clkI;
745  var sdClk = clk as SignalDescriptor;
746  if (sdClk == null)
747  clkI = ((SignalDescriptor)((PortDescriptor)clk).BoundSignal).Instance;
748  else
749  clkI = sdClk.Instance;

764  XILSFunction fnasm;
