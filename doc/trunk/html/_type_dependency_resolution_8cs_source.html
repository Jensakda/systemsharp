215  {
216  var fspec = callable as FunctionSpec;
217  if (fspec != null && fspec.GenericSysDOMRep != null)
218  {
219  var owner = (IPackageOrComponentDescriptor)fspec.GenericSysDOMRep.Owner;

252  static void ResolveTypeDependencies(this IComponentDescriptor component, DesignDescriptor design)
253  {
254  foreach (var child in component.GetChildComponents())

256  ResolveTypeDependencies(child, design);
257  }

259  var str = new StmtTypeResolver(design, component);

264  foreach (var port in component.GetPorts())

267  }
268  foreach (var signal in component.GetSignals())

270  str.RequireType(signal.ElementType, design);
271  }
272  foreach (var field in component.GetFields())
273  {
274  str.RequireType(field.Type, design);
