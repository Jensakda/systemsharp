122  internal void Run()
123  {
124  using (_ps)
125  {
126  bool started;
127  lock (_syncRoot)
128  {
129  started = _ps.Start();
130  if (started)
131  {
132  try
133  {
134  _name = _ps.ProcessName;
135  _ps.PriorityClass = ProcessPriorityClass.BelowNormal;
136  }
137  catch (Exception)
138  {
139  // The properties are only accessible if the process is running.
140  // We're actually in a race condition: It might happen that the
141  // process already exited when we access those properties. In
142  // this case, an exception will be thrown.
143  }
144  _state = EToolState.Running;
145  _pool.OnToolSpawned(this);
146  _ps.BeginOutputReadLine();
147  _ps.BeginErrorReadLine();
148  }
149  else
150  {
151  _state = EToolState.StartFailure;
152  _pool.OnFailedToolStart(this);
153  }
154  _isStarted.Set();
155  }
156  if (started)
157  {
158  _ps.WaitForExit();
159  lock (_syncRoot)
160  {
161  ExitCode = _ps.ExitCode;
162  _state = EToolState.Exited;
163  _pool.OnToolExited(this);
164  }
165  }
166  _isExited.Set();
167  }
