513  protected virtual AbstractStackState<TElem> Propagate(ILInstruction ili, AbstractStackState<TElem> pre)
514  {
515  int preDepth = pre.Depth;
516  AbstractStackState<TElem> post = _pmap[ili.Code](ili, pre);
517  int postDepth = post.Depth;

551  public virtual void Run()
552  {
553  Contract.Requires(!AnalysisDone);
554  Contract.Ensures(AnalysisDone);
555 
556  MethodFacts myFacts = FactUniverse.Instance.GetFacts(Method);
557  CFG = myFacts.CFG;
558 
559  _stackStates = new AbstractStackState<TElem>[CFG.Instructions.Count];
560  AbstractStackState<TElem> initial = CreateInitialStackState();
561  Queue<Tuple<AbstractStackState<TElem>, ILInstruction>> q = new Queue<Tuple<AbstractStackState<TElem>, ILInstruction>>();
562  q.Enqueue(Tuple.Create(initial, CFG.Instructions[0]));
563  int marshal = CFG.Instructions.Count - 1;
564  while (q.Count > 0)
565  {
566  Tuple<AbstractStackState<TElem>, ILInstruction> cur = q.Dequeue();
567  AbstractStackState<TElem> preState = cur.Item1;
568  ILInstruction ili = cur.Item2;
569  AbstractStackState<TElem> nextState = Propagate(ili, preState);
570  ILInstruction[] succs = CFG.GetSuccessorsOf(ili);
571  if (_stackStates[ili.Index] == null)
572  _stackStates[ili.Index] = nextState;
573  else
574  {
575  bool changed = Merge(_stackStates[ili.Index], nextState, out _stackStates[ili.Index]);
576  if (!changed)
577  continue;
578  }
579  foreach (ILInstruction nextILI in succs)
580  {
581  if (nextILI.Index < marshal) // skip pseudo ret at the end
582  q.Enqueue(Tuple.Create(_stackStates[ili.Index], nextILI));
583  }
584  }
585 
586  AnalysisDone = true;
587  }
588  }
