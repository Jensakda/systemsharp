178  public IEnumerable<IXILMapping> TryMap(ITransactionSite taSite, XILInstr instr, TypeDescriptor[] operandTypes, TypeDescriptor[] resultTypes)
179  {
180  var fu = taSite.Host;
181  BCU bcu = fu as BCU;
182  if (bcu != _host)
183  yield break;
184 
185  switch (instr.Name)
186  {
187  case InstructionCodes.Goto:
188  case InstructionCodes.BranchIfTrue:
189  case InstructionCodes.BranchIfFalse:
190  {
191  var target = (BranchLabel)instr.Operand;
192  switch (instr.Name)
193  {
194  case InstructionCodes.Goto:
195  yield return new GotoMapping(bcu, target);
196  yield break;
197 
198  case InstructionCodes.BranchIfTrue:
199  yield return new BranchIfMapping(bcu, target);
200  yield break;
201 
202  case InstructionCodes.BranchIfFalse:
203  yield return new BranchIfNotMapping(bcu, target);
204  yield break;
205 
206  default:
207  throw new NotImplementedException();
208  }
209  }
210 
211  default:
212  yield break;
213  }
214  }
