77  var convVar = needRead ? IntrinsicFunctions.Cast(lrVar, elemType.CILType, taSite._dataOut.ElementType) : null;

80  var lr0 = LiteralReference.CreateConstant((StdLogicVector)"0");

86  BodyBuilder.Store(taSite._literal, Expression.Equal(lrDataIn, lr1));
