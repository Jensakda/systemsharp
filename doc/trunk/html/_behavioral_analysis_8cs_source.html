230  private void Resolve(MethodCallInfo mci)
231  {
232  Function genericImpl;
233  var md = ConstructMethodDescriptor(mci, false);
234  if (md == null)
235  return;
236  object instance = mci.Instance;
237  bool hasThis = md.Method.CallingConvention.HasFlag(CallingConventions.HasThis);
238  if (hasThis)
239  {
240  if (instance != null)
241  {
242  var owner = instance as Component;
243  if (owner == null)
244  {
245  ReportError("Method " + md.Method.Name + " is not declared inside a component.");
246  }
247  else
248  {
249  md.Owner = owner.Descriptor;
250  }
251  }
252  else
253  {
254  ReportError("Method " + md.Method.Name + ": unable to resolve declaring component instance.");
255  }
256  }
257  else
258  {
259  md.Owner = GetPackageDescriptor(md.Method.DeclaringType);
260  }
261  var key = new MethodKey(md);
262  if (_methodMap.TryGetValue(key, out genericImpl))
263  {
264  md.Implementation = genericImpl;
265  }
266  else
267  {
268  DecompileAndEnqueueCallees(md, instance, md.ArgValueSamples, mci);
269  System.Diagnostics.Debug.Assert(md.Implementation != null);
270  _methodMap[key] = md.Implementation;
271  }
272 
273  System.Diagnostics.Debug.Assert(md.Implementation != null);
274  if (md.Owner != null)
275  md = md.Owner.Canonicalize(md);
276 
277  var mds = ConstructMethodDescriptor(mci, true);
278  mds.Owner = md.Owner;
279  DecompileAndEnqueueCallees(mds, instance, mds.ArgValueSamples, mci);
280  System.Diagnostics.Debug.Assert(mds.Implementation != null);
281  mci.Resolve(md, mds);
282  }
