370  virtual protected void OnAnalysis()
371  {
372  Type type = GetType();
373  var fis = FindFields(type);
374  foreach (FieldInfo fi in fis)
375  {
376  // Skip backing fields of properties (these are mapped to ports)
377  // There is currently no elegant way of doing this. The only possibility is to check whether the
378  // field name contains a "magic" character which is usually not allowed as an identifier name.
379  if (fi.Name.Contains('<'))
380  continue;
381 
382  object fvalue = fi.GetValue(this);
383  IContainmentImplementor ici = fvalue as IContainmentImplementor;
384  if (ici != null)
385  {
386  ici.SetOwner(Descriptor, fi, IndexSpec.Empty);
387  }
388  }
389 
390  var pis = FindProperties(type);
391  foreach (PropertyInfo pi in pis)
392  {
393  Type propType = pi.PropertyType;
394  RewriteDeclaration rwDecl = (RewriteDeclaration)Attribute.GetCustomAttribute(
395  propType, typeof(RewriteDeclaration), true);
396  if (rwDecl != null)
397  rwDecl.ImplementDeclaration(Descriptor, pi);
398  }
399  }
