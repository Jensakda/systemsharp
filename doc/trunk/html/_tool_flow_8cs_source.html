183  private ProcessPool.ToolBatch CreateBatch(EFlowStep steps)
184  {
185  var bat = ProcessPool.Instance.CreateBatch();
186  if (steps.HasFlag(EFlowStep.XST))
187  XST.SaveToXSTScriptAndAddToBatch(Project, bat, XSTScriptPath, XSTLogPath);
188  if (steps.HasFlag(EFlowStep.NGDBuild))
189  NGDBuild.AddToBatch(Project, bat);
190  if (steps.HasFlag(EFlowStep.Map))
191  Map.AddToBatch(Project, bat);
192  if (steps.HasFlag(EFlowStep.PAR))
193  PAR.AddToBatch(Project, bat);
194  if (steps.HasFlag(EFlowStep.TRCE))
195  TRCE.AddToBatch(Project, bat);
196  return bat;
197  }
