175  protected InstructionDependency[] RemapPreds(InstructionDependency[] preds)
176  {
177  var result = new List<InstructionDependency>();
178  foreach (var pred in preds)
179  {
180  int prev, cur;
181  if (!_remap.TryGetValue(pred.PredIndex, out cur))
182  continue;
183  if (!_remap.TryGetValue(pred.PredIndex - 1, out prev))
184  prev = cur - 1;
185  for (int i = prev + 1; i <= cur; i++)
