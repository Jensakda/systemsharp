59  public int FindSet(int elementId)
60  {
61  Contract.Requires(elementId < ElementCount);
62  Contract.Requires(elementId >= 0);
63 
64  Node curNode;
65 
66  // Find the root element that represents the set which `elementId` belongs to
67  curNode = m_nodes[elementId];
68  while (curNode.Parent != null)
69  curNode = curNode.Parent;
70  Node root = curNode;
71 
72  // Walk to the root, updating the parents of `elementId`. Make those elements the direct
73  // children of `root`. This optimizes the tree for future FindSet invokations.
74  curNode = m_nodes[elementId];
75  while (curNode != root)
76  {
77  Node next = curNode.Parent;
78  curNode.Parent = root;
79  curNode = next;
80  }
81 
82  return root.Index;
83  }

90  public void Union(int setId1, int setId2)
91  {
92  if (setId1 >= m_elementCount)
93  throw new ArgumentOutOfRangeException("setId1");
94  if (setId2 >= m_elementCount)
95  throw new ArgumentOutOfRangeException("setId2");
96 
97  if (setId1 == setId2)
98  return; // already unioned
99 
100  Node set1 = m_nodes[setId1];
101  Node set2 = m_nodes[setId2];
102 
103  // Determine which node representing a set has a higher rank. The node with the higher rank is
104  // likely to have a bigger subtree so in order to better balance the tree representing the
105  // union, the node with the higher rank is made the parent of the one with the lower rank and
106  // not the other way around.
107  if (set1.Rank > set2.Rank)
108  set2.Parent = set1;
109  else if (set1.Rank < set2.Rank)
110  set1.Parent = set2;
111  else // set1.Rank == set2.Rank
112  {
113  set2.Parent = set1;
114  ++set1.Rank; // update rank
115  }
116 
117  // Since two sets have fused into one, there is now one less set so update the set count.
118  --m_setCount;
119  }

126  public void AddElements(int addCount)
127  {
128  if (addCount < 0)
129  throw new ArgumentOutOfRangeException("addCount");
130 
131  // insert and initialize the specified number of element nodes to the end of the `m_nodes` array
132  for (int i = m_elementCount; i < m_elementCount + addCount; ++i)
133  {
134  Node newNode = new Node();
135  newNode.Parent = null;
136  newNode.Index = i;
137  newNode.Rank = 0;
138  m_nodes.Add(newNode);
139  }
140 
141  // update element and set counts
142  m_elementCount += addCount;
