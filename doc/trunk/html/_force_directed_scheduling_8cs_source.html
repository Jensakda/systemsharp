308  private double GetStorageDG(T instr)
309  {
310  long myAsap = Adapter.ASAPIndex[instr];
311  long myAlap = Adapter.ALAPIndex[instr];
312  double sum = 0.0;
313  int count = 0;
314  foreach (int result in Adapter.Results[instr])
315  {
316  var consumers = _operandConsumers[result];
317  long maxAsap = ASAP(consumers[0]);
318  long maxAlap = ALAP(consumers[0]);
319  for (int i = 1; i < consumers.Count; i++)
320  {
321  maxAsap = Math.Max(maxAsap, ASAP(consumers[i]));
322  maxAlap = Math.Max(maxAlap, ALAP(consumers[i]));
323  }
324  long asapLife = maxAsap - myAsap;
325  long alapLife = maxAlap - myAlap;
326  long maxLife = maxAlap - myAsap;
327  double avgLife = (asapLife + alapLife + maxLife) / 3.0;
328  long overlap = 0;
329  if (myAlap < maxAsap)
330  overlap = maxAsap - myAlap;
331  if (maxLife - overlap > double.Epsilon)
332  {
333  double storageDG = (avgLife - overlap) / (maxLife - overlap);
334  sum += storageDG;
335  count++;
336  }
337  }
338  if (count > 0)
339  return sum / count;
340  else
341  return 0.0;
342  }

344  private double GetForce(T instr, long nt, long nb)
345  {
346  if (IsScheduled(instr))
347  {
348  long cstep = Adapter.FDSIndex[instr];
349  if (cstep < nt || cstep > nb)
350  {
351  // This extension is required for proper support of multi-cycle operations:
352  // If an instruction is already scheduled and the query interval does not
353  // intersect with its c-step, return an "infinite" force to prevent predecessor/
354  // successor instructions being scheduled "too close" to the instruction.
355  return double.PositiveInfinity;
356  }
357  }
358  object op = Adapter.IClass[instr];
359  double sum = 0.0;

361  DG dg = _dg[op];
362  for (long cstep = nt; cstep <= nb; cstep++)
363  {
364  sum += dg[cstep] / h;
365  }

400  private void Recompute(T instr)
401  {
402  var s = new Stack<T>(
403  Adapter.Preds[instr].Select(_ => _.Task)
404  .Concat(Adapter.Succs[instr].Select(_ => _.Task)));
405  var l = new List<T>();
406  l.Add(instr);
407  while (s.Any())
408  {
409  T cur = s.Pop();
410 
411  long asap = ASAP(cur);
412  long alap = ALAP(cur);
413  long oldAsap = asap;
414  long oldAlap = alap;
415  foreach (var pred in Adapter.Preds[cur])
416  {
417  asap = Math.Max(asap, ASAP(pred.Task) + pred.MinDelay);
418  alap = Math.Min(alap, ALAP(pred.Task) + pred.MaxDelay);
419  }
420  //long curlat = Adapter.Latency[cur];
421  foreach (var succ in Adapter.Succs[cur])
422  {
423  alap = Math.Min(alap, ALAP(succ.Task) - succ.MinDelay);
424  asap = Math.Max(asap, ASAP(succ.Task) - succ.MaxDelay);
425  }
426  if (asap > alap)
427  throw new NotSchedulableException();
428  if (asap != oldAsap || alap != oldAlap)
429  {
430  object op = Adapter.IClass[cur];
431  DG dg = _dg[op];
432  dg.Unconsume(cur);
433  SetASAP(cur, asap);
434  SetALAP(cur, alap);
435  dg.Consume(cur);
436  l.Add(cur);
437  foreach (var pred in Adapter.Preds[cur])
438  {
439  s.Push(pred.Task);
440  }
441  foreach (var succ in Adapter.Succs[cur])
442  {
443  s.Push(succ.Task);
444  }
445  }
446  }
447  ComputeIDs(l);
448  }

465  private void Schedule()
466  {
467  var unscheduled = new LinkedList<T>(Instructions);
468  while (unscheduled.Any())
469  {
470  var bestNode = unscheduled.First;
471  T bestInstr = bestNode.Value;
472  long bestCStep = ASAP(bestInstr);
473  double minForce = GetTotalForce(bestInstr, bestCStep);
474  var curNode = bestNode;
475  while (curNode != unscheduled.Last)
476  {
477  T instr = curNode.Value;
478  long asap = ASAP(instr);
479  long alap = ALAP(instr);
480 
481  if (asap == alap)
482  // there will never be a different option for this instruction
483  // => we can abbreviate the process a little bit
484  break;
485 
486  if (asap > alap)
487  throw new NotSchedulableException();
488 
489  for (long cstep = asap; cstep <= alap; cstep++)
490  {
491  double force = GetTotalForce(instr, cstep);
492  if (force < minForce)
493  {
494  bestNode = curNode;
495  bestInstr = instr;
496  bestCStep = cstep;
497  minForce = force;
498  }
499  }
500  curNode = curNode.Next;
501  }
502  if (double.IsPositiveInfinity(minForce))
503  throw new NotSchedulableException();
504 
505  Pin(bestInstr, bestCStep);
506  unscheduled.Remove(bestNode);
507  }
508  }
