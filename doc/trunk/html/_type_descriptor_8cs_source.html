519  private string CheckStatic()
520  {
521  if (HasIntrinsicTypeOverride)
522  return null;
523 
524  if (CILType.IsArray)
525  {
526  TypeDescriptor tdref = GetElement0Type();
527  string innerCheck = tdref.CheckStatic();
528  if (innerCheck != null)
529  return innerCheck;
530  if (_sample == null)
531  return "No sample - not able to determine whether type descriptor is static";
532  Array array = (Array)_sample;
533  int[] indices = new int[array.Rank];
534  do
535  {
536  object elem = array.GetValue(indices);
537  if (elem == null)
538  return "Null array element";
539 
540  TypeDescriptor td = new TypeDescriptor(elem);
541  if (!tdref.Equals(td))
542  return "Different element types in array";
543 
544  int dim;
545  for (dim = 0; dim < indices.Length; dim++)
546  {
547  ++indices[dim];
548  if (indices[dim] < array.GetLength(dim))
549  break;
550  indices[dim] = 0;
551  }
552  if (dim == indices.Length)
553  break;
554  }
555  while (true);
556 
557  return null;
558  }
559  else
560  {
561  if (!CILType.IsValueType && !CILType.IsPrimitive)
562  return "Type is neither primitive nor a value type";
563 
564  TypeDescriptor[] deps = GetDependentTypes();
565  foreach (TypeDescriptor dep in deps)
566  {
567  string innerCheck = dep.CheckStatic();
568  if (innerCheck != null)
569  return innerCheck;
570  }
571 
572  return null;
573  }
