191  public void CreateControlpath(FlowMatrix flowSpec, string procName)
192  {
193  int ncsteps = flowSpec.NumCSteps;
194  string report = _mcd.ComputeEncoding(flowSpec, _maxSelWidth);
195  _curCW = (SLVSignal)_binder.GetSignal<StdLogicVector>(EPortUsage.Default, "CurCW", null,
196  StdLogicVector._0s(_mcd.CWWidth));
197  var clkInst = _binder.GetSignal<StdLogic>(EPortUsage.Clock, "Clk", null, '0');
198  if (_staged)
199  _mcd.CreateStagedDecoder(_binder, _curCW, (SLSignal)clkInst, _registered);
200  else
201  _mcd.CreateDecoder(_binder, _curCW);
202  CreateROM(ncsteps);
203  for (int cstep = 0; cstep < ncsteps; cstep++)
204  {
205  var cw = _mcd.Encode(cstep, flowSpec.GetFlow(cstep));
206  _romIf.PreWrite(StdLogicVector.FromUInt((uint)cstep, _pc.Size), cw);
207  }
208 
209  var syncTempl = new SyncTemplate(this);
210  var syncFunc = syncTempl.GetAlgorithm();
211  _binder.CreateProcess(SystemSharp.Components.Process.EProcessKind.Triggered, syncFunc, clkInst.Descriptor);
212 
213  _host.Descriptor.GetDocumentation().Documents.Add(new Document(procName + "_HMA_report.txt", report));
