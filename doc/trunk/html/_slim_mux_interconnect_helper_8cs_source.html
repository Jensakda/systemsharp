200  bool isCycle = source == sink ||

202  _succs[sink].Select(p => _pipes[p].sink).Contains(source);
203 
204  //Debug.Assert(source != sink);

206  //Debug.Assert(!_succs[sink].Select(p => _pipes[p].sink).Contains(source));
207 
208  if (isCycle)
209  {
210  string before = GetInterconnectGraphForDotty();
211  File.WriteAllText("SlimMuxBugReport_before.dotty", before);
212  }
213 
214  // Disallow redundant pipes of delay 0

375  public bool[,] ComputePipeEnMatrix(int numCsteps)
376  {
377  var result = new bool[numCsteps, _pipes.Count];
378  var q = new LinkedList<long>();
379  var ts = new SortedSet<long>();
380  for (int pipe = 0; pipe < _pipes.Count; pipe++)
381  {
382  int capa = CalcPipeCapacity(pipe);
383  long delay = _pipes[pipe].delay;
384 
385  if (delay == 0)
386  continue;
387 
388  var reservations = _resTable[pipe];
389  var restimes = reservations.Keys.OrderBy(k => k);
390 
391  bool pass = false;
392 
393  do
394  {
395  q.Clear();
396  ts.Clear();
397  foreach (long time in restimes)
398  {
399  while (q.Count > 0 && q.First.Value < time)
400  {
401  ts.Add(q.First.Value);
402  result[q.First.Value, pipe] = true;
403  q.RemoveFirst();
404  }
405  ts.Add(time);
406  ts.Add(time + delay);
407  Debug.Assert(time + delay < numCsteps);
408  result[time, pipe] = true;
409  if (q.Count > 0)
410  q.RemoveFirst();
411  long post = delay - capa + q.Count + 1;
412  for (long strobe = post; strobe < delay; strobe++)
413  {
414  long mark = time + strobe;
415  if (q.Count == 0 || q.Last.Value < mark)
416  q.AddLast(mark);
417  }
418  }
419  while (q.Count > 0)
420  {
421  result[q.First.Value, pipe] = true;
422  ts.Add(q.First.Value);
423  q.RemoveFirst();
424  }
425 
426  // Verify
427  var afetchtimes = restimes.Select(t => t + delay).ToArray();
428  int nf = 0;
429  var fifo = new LinkedList<long>();
430  pass = true;
431  foreach (long time in ts)
432  {
433  Debug.Assert(nf < afetchtimes.Length);
434  if (afetchtimes[nf] == time)
435  {
436  Debug.Assert(fifo.Count == capa);
437  //Debug.Assert(fifo.First.Value == time - delay);
438  if (fifo.First.Value != time - delay)
439  {
440  pass = false;
441  ++capa;
442  break;
443  }
444  nf++;
445  }
446  Debug.Assert(time < numCsteps);
447  if (result[time, pipe])
448  {
449  fifo.AddLast(time);
450  if (fifo.Count > capa)
451  fifo.RemoveFirst();
452  }
453  }
454 
455  Debug.Assert(capa <= delay);
456 
457  } while (!pass);
458 
459  _pipes[pipe].capacity = capa;
460  _pipes[pipe].useEn = capa < delay;
461  }
462  return result;
463  }

465  public IEnumerable<ParFlow> ToFlow(int numCsteps, ParFlow neutralFlow, bool[,] pipeEnMatrix)
466  {
467  int muxCount = _pipeInSignals.Count;
468  int[,] flowMatrix = new int[numCsteps, muxCount];
469 
470  // 1st pass: set some arbitrary MUX selection
471  for (int i = 0; i < numCsteps; i++)
472  {
473  for (int j = 0; j < muxCount; j++)
474  {
475  if (_preds[j].Any())
476  flowMatrix[i, j] = _preds[j].First();
477  else
478  flowMatrix[i, j] = -1;
479  }
480  }
481 
482  // 2nd pass: reset MUX selection whenever neutral flow requires
483  // some value transfer which is not "don't care"
484  foreach (var flow in neutralFlow.Flows)
485  {
486  if (FlowMatrix.IsDontCareFlow(flow) ||
487  !_signal2Idx.IsCached(flow.Target))
488  continue;
489 
490  int idx = _signal2Idx[flow.Target];
491  for (int i = 0; i < numCsteps; i++)
492  {
493  flowMatrix[i, idx] = -1;
494  }
495  }
496 
497  // 3rd pass: transfer MUX reservations to matrix
498  for (int i = 0; i < _resTable.Count; i++)
499  {
500  var rmap = _resTable[i];
501  var pi = _pipes[i];
502  foreach (var kvp in rmap)
503  {
504  flowMatrix[kvp.Key + pi.delay, pi.sink] = i;
505  }
506  }
507 
508  var pipeen = pipeEnMatrix;
509 
510  // last pass: convert to flows
511  for (int i = 0; i < numCsteps; i++)
512  {
513  var flows = new List<Flow>();
514  for (int j = 0; j < muxCount; j++)
515  {
516  int k = flowMatrix[i, j];
517  if (k >= 0)
518  {
519  flows.Add(
520  new SignalFlow(
521  _pipeOutSignals[k].ToSignalRef(SignalRef.EReferencedProperty.Cur),
522  _pipeInSignals[j].ToSignalRef(SignalRef.EReferencedProperty.Next)));
523  }
524  }
525  for (int pipe = 0; pipe < _pipes.Count; pipe++)
526  {
527  if (!_pipes[pipe].useEn)
528  continue;
529  flows.Add(
530  new ValueFlow(
531  pipeen[i, pipe] ? StdLogic._1 : StdLogic._0,
532  _pipeEnSignals[pipe].ToSignalRef(SignalRef.EReferencedProperty.Next)));
533  }
534  yield return new ParFlow(flows);
535  }
536  }

538  private int CalcPipeCapacity(int pipe)
539  {
540  Contract.Requires<ArgumentException>(pipe >= 0);
541 
542  long delay = _pipes[pipe].delay;
543  var reservations = _resTable[pipe];
544  var restimes = reservations.Keys.OrderBy(k => k);
545  var q = new LinkedList<long>();
546  int capa = 0;
547  foreach (long time in restimes)
548  {
549  q.AddLast(time);
550  long kicktime = time - delay;
551  if (q.First.Value <= kicktime)
552  q.RemoveFirst();
553  if (q.Count > capa)
554  capa = q.Count;
555  }
556  Debug.Assert(capa <= delay);
