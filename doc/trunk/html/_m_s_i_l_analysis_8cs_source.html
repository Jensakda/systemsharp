249  private IEnumerable<ReferenceInfo> HandleCall(ILInstruction ili, ControlFlowGraph<ILInstruction> cfg)
250  {
251  MethodBase mb = (MethodBase)ili.Operand;
252  List<ReferenceInfo.EMode> modes = new List<ReferenceInfo.EMode>();
253  if (!mb.IsStatic && !mb.IsConstructor)
254  {
255  modes.Add(ReferenceInfo.EMode.Read | ReferenceInfo.EMode.Write);
256  }
257  ParameterInfo[] pis = mb.GetParameters();
258  foreach (ParameterInfo pi in pis)
259  {
260  if (pi.ParameterType.IsByRef)
261  {
262  if (pi.IsOut)
263  modes.Add(ReferenceInfo.EMode.Write);
264  else
265  modes.Add(ReferenceInfo.EMode.Read |
266  ReferenceInfo.EMode.Write);
267  }
268  else
269  {
270  modes.Add(ReferenceInfo.EMode.NoAccess);
271  }
272  }
273  if (ili.Code.Equals(OpCodes.Calli))
274  {
275  modes.Add(ReferenceInfo.EMode.Read | ReferenceInfo.EMode.Write);
276  }
277  return modes.Select((m, i) => new ReferenceInfo(m, ReferenceInfo.EKind.Indirect,
278  StackInfluenceAnalysis.GetStackElementDefinitions(ili.Index, i - modes.Count + 1, (MethodCode)cfg)));
279  }

572  private void ComputeStackDepths(int startIndex, int stack)
573  {
574  int index = startIndex;
575  int count = Instructions.Count;
576  while (index < count)
577  {
578  ILInstruction ili = Instructions[index];
579  int npop, npush;
580  StackInfluenceAnalysis.GetStackBilance(ili, Method, out npop, out npush);
581  int delta = npush - npop;
582 
583  _preStackDepth[ili.Index] = stack;
584  stack += delta;
585  if (stack < 0)
586  throw new InvalidOperationException("The MSIL code is ill-formed (stack underrun)");
587 
588  if (_postStackDepth[ili.Index] < 0)
589  _postStackDepth[ili.Index] = stack;
590  else if (_postStackDepth[ili.Index] != stack)
591  throw new InvalidOperationException("The MSIL code is ill-formed (inconsistent stack behavior)");
592  else
593  return;
594 
595  ILInstruction[] succs = GetSuccessorsOf(ili);
596  if (succs.Length == 0)
597  return;
598  else if (succs.Length == 1)
599  {
600  index = succs[0].Index;
601  if (index == InstructionInfo.Marshal.Index)
602  return;
603  }
604  else
605  {
606  foreach (ILInstruction succ in succs)
607  {
608  if (succ.Index != InstructionInfo.Marshal.Index)
609  ComputeStackDepths(succ.Index, stack);
610  }
611  return;
612  }
613  }
614  }

619  _postStackDepth = new int[Instructions.Count];

621  _postStackDepth[i] = -1;
622  ComputeStackDepths(EntryCB.StartIndex, 0);
623  for (int i = 0; i < _postStackDepth.Length; i++)
624  {
625  if (_postStackDepth[i] < 0)
626  _postStackDepth[i] = 0;
627  }
