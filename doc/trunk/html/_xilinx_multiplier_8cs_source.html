141  public IEnumerable<TAVerb> Multiply(ISignalSource<StdLogicVector> a, ISignalSource<StdLogicVector> b, ISignalSink<StdLogicVector> r)
142  {
143  IProcess action = _host.A.Dual.Drive(a).Par(
144  _host.B.Dual.Drive(b));
145  if (_host.HasCE)
146  action = action.Par(_host.CE.Dual.Stick<StdLogic>('1'));
147  if (_host.HasSCLR)
148  action = action.Par(_host.SCLR.Dual.Stick<StdLogic>('0'));
149  if (_host.PipeStages == 0)
150  {
151  action = action.Par(r.Comb.Connect(_host.P.Dual.AsSignalSource()));
152  yield return Verb(ETVMode.Locked, action);
153  }
154  else
155  {
156  yield return Verb(ETVMode.Locked, action);
157  for (int i = 1; i < _host.PipeStages; i++)
158  yield return Verb(ETVMode.Shared);
159  yield return Verb(ETVMode.Shared, r.Comb.Connect(_host.P.Dual.AsSignalSource()));

638  public IXILMapping TryMapOne(ITransactionSite taSite, XILInstr instr, TypeDescriptor[] operandTypes, TypeDescriptor[] resultTypes, bool swap)
639  {
640  var fu = taSite.Host;
641  XilinxMultiplier mul = fu as XilinxMultiplier;
642  if (mul == null)
643  return null;
644 
645  if (instr.Name != InstructionCodes.Mul)
646  return null;
647 
648  if (mul.MultType != XilinxMultiplier.EMultiplierType.ParallelMultiplier)
649  return null;
650 
651  FixFormat fmta, fmtb, fmtr;
652  fmta = GetFixFormat(operandTypes[0]);
653  fmtb = GetFixFormat(operandTypes[1]);
654  fmtr = GetFixFormat(resultTypes[0]);
655  if (fmta == null || fmtb == null || fmtr == null)
656  return null;
657 
658  bool expectSigned = fmta.IsSigned || fmtb.IsSigned;
659  if (expectSigned != fmtr.IsSigned)
660  return null;
661 
662  int expectedHigh = fmtr.IntWidth + fmta.FracWidth + fmtb.FracWidth - 1;
663  int expectedLow = fmta.FracWidth + fmtb.FracWidth - fmtr.FracWidth;
664 
665  if (fmta.TotalWidth != mul.PortAWidth ||
666  fmtb.TotalWidth != mul.PortBWidth ||
667  expectedHigh != mul.OutputWidthHigh ||
668  expectedLow != mul.OutputWidthLow)
669  return null;
670 
671  return new XILMapping(mul, swap);
672  }
