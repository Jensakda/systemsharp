170  private static FixFormat Equalize(SFix a, SFix b, out Signed an, out Signed bn)
171  {
172  an = a._value;
173  bn = b._value;
174  FixFormat dfmt;
175  switch (DesignContext.Instance.FixPoint.ArithSizingMode)
176  {
177  case EArithSizingMode.Safe:
178  dfmt = new FixFormat(true,
179  Math.Max(a.Format.IntWidth, b.Format.IntWidth) + 1,
180  Math.Max(a.Format.FracWidth, b.Format.FracWidth));
181  an = an.Resize(dfmt.TotalWidth - 1);
182  bn = bn.Resize(dfmt.TotalWidth - 1);
183  if (a.Format.FracWidth > b.Format.FracWidth)
184  bn = bn << (int)(a.Format.FracWidth - b.Format.FracWidth);
185  else if (b.Format.FracWidth > a.Format.FracWidth)
186  an = an << (int)(b.Format.FracWidth - a.Format.FracWidth);
187  return dfmt;
188 
189  case EArithSizingMode.VHDLCompliant:
190  dfmt = new FixFormat(true,
191  Math.Max(a.Format.IntWidth, b.Format.IntWidth) + 1,
192  Math.Max(a.Format.FracWidth, b.Format.FracWidth));
193  an = an.Resize(dfmt.TotalWidth);
194  bn = bn.Resize(dfmt.TotalWidth);
195  if (a.Format.FracWidth > b.Format.FracWidth)
196  bn = bn << (int)(a.Format.FracWidth - b.Format.FracWidth);
197  else if (b.Format.FracWidth > a.Format.FracWidth)
198  an = an << (int)(b.Format.FracWidth - a.Format.FracWidth);
199  return dfmt;
200 
201  case EArithSizingMode.InSizeIsOutSize:
202  dfmt = a.Format;
203  bn = bn.Resize(dfmt.TotalWidth);
204  if (a.Format.FracWidth > b.Format.FracWidth)
