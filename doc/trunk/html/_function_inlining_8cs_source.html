180  private bool TryInline(ICallable callable, Expression[] args, out IStorableLiteral retv)
181  {
182  FunctionSpec fspec = callable as FunctionSpec;
183  retv = null;
184  if (fspec != null)
185  {
186  // Do not inline intrinsic functions
187  if (fspec.IntrinsicRep != null)
188  return false;
189 
190  var md = fspec.SpecialSysDOMRep;
191  if (md != null)
192  {
193  var fun = md.Implementation;
194 
195  InlineVerifier.CheckAllLocalsDeclared(fun);
196 
197  // Recursion?
198  if (_funStack.Contains(fun))
199  return false;
200 
201  if (fun.OutputVariables.Count == 0)
202  {
203  if (!fspec.ResultType.CILType.Equals(typeof(void)))
204  {
205  string name = _sim.GetUniqueName("$ret", new object());
206  IStorableLiteral rv = new Variable(fspec.ResultType)
207  {
208  Name = name
209  };
210  retv = _locals[Tuple.Create(fun, rv)];
211  }
212  }
213  else if (fun.OutputVariables.Count == 1)
214  {
215  retv = _locals[Tuple.Create(fun, fun.OutputVariables[0])];
216  }
217  else
218  {
219  throw new NotSupportedException();
220  }
221 
222  // account for possible "this" argument
223  int j = args.Length - fun.InputVariables.Count;
224 
225  for (int i = j; i < args.Length; i++)
226  {
227  Expression rhs = args[i].Transform(this);
228  var lhs = _locals[Tuple.Create(fun, fun.InputVariables[i - j])];
229  Store(lhs, rhs);
230  }
231  if (retv != null)
232  _retStack.Push(retv);
233  _funStack.Push(fun);
234  _getOutStack.Push(new List<GotoStatement>());
235  fun.Body.Accept(this);
236  _funStack.Pop();
237  if (retv != null)
238  _retStack.Pop();
239  Nop();
240  foreach (var getOut in _getOutStack.Pop())
241  {
242  getOut.Target = LastStatement;
243  }
244 
245  _inlinedFunctions.Add(fun);
246 
247  return true;
248  }
249  else
250  {
251  //System.Diagnostics.Debug.Assert(false, "function not decompiled - bug?");
252  }
253  }
254  return false;
255  }
