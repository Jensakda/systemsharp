516  public override StdLogicVector[] SerializeInstance(object instance)
517  {
518  Array array = (Array)instance;
519  IEnumerable<object> elements = array.Cast<object>();
520  int rank = Strides.Length;
521  int[] curSrcIndex = new int[rank];
522  int curDstIndex = 0;
523  int curSubIndex = 0;
524  StdLogicVector[] result = Enumerable.Repeat(
525  StdLogicVector._0s(WordSize), array.GetLength(0) * (int)Strides[0])
526  .ToArray();
527  foreach (object element in elements)
528  {
529  StdLogicVector[] elementData = ElementLayout.SerializeInstance(element);
530  curDstIndex = 0;
531  for (int i = 0; i < rank; i++)
532  {
533  curDstIndex += (int)Strides[i] * curSrcIndex[i];
534  }
535  if (SubStride == 0)
536  {
537  Array.Copy(elementData, 0, result, curDstIndex, elementData.Length);
538  }
539  else
540  {
541  Debug.Assert(elementData.Length == 1);
542  result[curDstIndex][(curSubIndex+1) * (int)SubStride - 1, curSubIndex * (int)SubStride] =
543  elementData[0][(int)SubStride - 1, 0];
544  if (++curSubIndex == ElementsPerWord)
545  {
546  curSubIndex = 0;
547  }
548  else
549  {
550  continue;
551  }
552  }
553  for (int i = rank - 1; i >= 0; i--)
554  {
555  if (++curSrcIndex[i] == array.GetLength(i))
556  curSrcIndex[i] = 0;
557  else
558  break;
559  }

597  public static MemoryLayout Layout(TypeDescriptor td, IMarshalInfo info)
598  {
599  Type type = td.CILType;
600  ISerializer ser = SLVSerializable.TryGetSerializer(type);
601  if (ser != null)
602  {
603  object sample = td.GetSampleInstance();
604  return CreatePrimLayout(ser.Serialize(sample).Size, td, info);
605  }
606  if (ser == null && td.HasIntrinsicTypeOverride)
607  throw new InvalidOperationException("Type " + type.Name + " has intrinsic type override but no serializer");
608  if (type.IsEnum)
609  {
610  return new EnumMemoryLayout(td, info);
611  }
612  if (type.IsArray)
613  {
614  td.AssertStatic();
615  TypeDescriptor elemTd = td.Element0Type;
616  MemoryLayout elemLayout = Layout(elemTd, info);
617  ulong subStride = elemLayout.SizeInBits;
618  if (subStride == 0)
619  {
620  return new EmptyMemoryLayout(td)
621  {
622  Size = 0,
623  SizeInBits = 0
624  };
625  }
626  ulong[] strides = new ulong[type.GetArrayRank()];
627  ulong elemsPerWord = (ulong)info.WordSize / elemLayout.SizeInBits;
628  ulong wordsPerElem = elemLayout.Size;
629  if (elemsPerWord > 1)
630  {
631  if (info.UseArraySubWordAlignment)
632  {
633  if (info.UseArrayDimPow2Alignment)
634  {
635  elemsPerWord = MathExt.FloorPow2(elemsPerWord);
636  subStride = info.WordSize / elemsPerWord;
637  }
638  }
639  else
640  {
641  elemsPerWord = 1;
642  }
643  }
644  ulong dimSize = (ulong)(int)td.TypeParams.Last();
645  ulong dimWords;
646  if (elemsPerWord <= 1)
647  {
648  subStride = 0;
649  if (info.UseArrayDimPow2Alignment)
650  wordsPerElem = MathExt.CeilPow2(wordsPerElem);
651  dimWords = wordsPerElem * dimSize;
652  }
653  else
654  {
655  wordsPerElem = 0;
656  dimWords = (dimSize + elemsPerWord - 1) / elemsPerWord;
657  }
658  strides[strides.Length-1] = wordsPerElem;
659  for (int i = strides.Length-2; i >= 0; i--)
660  {
661  if (info.UseArrayDimPow2Alignment)
662  dimWords = MathExt.CeilPow2(dimWords);
663  strides[i] = dimWords;
664  dimSize = (ulong)(int)td.TypeParams[i];
665  dimWords *= dimSize;
666  }
667  return new ArrayMemoryLayout(td, info.WordSize, strides, subStride, (uint)elemsPerWord, (uint)wordsPerElem, elemLayout)
668  {
669  Size = dimWords,
670  SizeInBits = dimWords * info.WordSize
671  };
672  }
673  if (type.IsValueType && !type.IsPrimitive)
674  {
675  StructMemoryLayout ml = new StructMemoryLayout(td);
676  FieldInfo[] fields = type.GetFields(
677  BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance);
678  ulong offset = 0;
679  foreach (FieldInfo field in fields)
680  {
681  TypeDescriptor fieldTd = td.GetFieldType(field);
682  MemoryLayout fieldLayout = Layout(fieldTd, info);
683  FieldLocation fieldLoc = new FieldLocation(field, fieldLayout, offset);
684  ml.AddLocation(fieldLoc);
685  offset += fieldLayout.Size;
686  }
687  ml.Size = offset;
688  ml.SizeInBits = offset * info.WordSize;
689  return ml;
690  }
691 
692  throw new InvalidOperationException("Unable to create data layout for type " + type.Name);
693  }
