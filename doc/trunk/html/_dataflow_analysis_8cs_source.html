431  private void IterateToFixPoint()
432  {
433  List<Ti> instrs = Code.Instructions;
434  _preConds = new VarAssignmentSet[instrs.Count];
435  _postConds = new VarAssignmentSet[instrs.Count];
436  _refPoints = new int[instrs.Count];
437  _refUses = new Dictionary<int, List<int>>();
438  _inhibitElimination = new bool[_numLocals];
439  Queue<int> q = new Queue<int>();
440  _loadPoints = new HashSet<int>[_numLocals];
441  _storePoints = new HashSet<int>[_numLocals];
442  for (int i = 0; i < _numLocals; i++)
443  {
444  if (Code.IsLocalPinned(i))
445  InhitbitElimination(i);
446  _loadPoints[i] = new HashSet<int>();
447  _storePoints[i] = new HashSet<int>();
448  }
449  for (int i = 0; i < instrs.Count; i++)
450  {
451  _preConds[i] = new VarAssignmentSet(_numLocals);
452  _postConds[i] = new VarAssignmentSet(_numLocals);
453  _refPoints[i] = -1;
454  }
455  int startIndex = Code.EntryCB.StartIndex;
456  _preConds[startIndex].SetInitial();
457  q.Enqueue(startIndex);
458 
459  IExtendedInstructionInfo<Ti> xinfo = InstructionInfo as IExtendedInstructionInfo<Ti>;
460  while (q.Count > 0)
461  {
462  int iidx = q.Dequeue();
463  _curILI = Code.Instructions[iidx];
464  _curPreCond = _preConds[iidx];
465  _curPostCond = _postConds[iidx];
466  _curPostCond.Merge(_curPreCond);
467  ProcessInstruction();
468  if (xinfo != null)
469  {
470  ProcessInstructionExtended();
471  }
472  Ti[] succs = Code.GetSuccessorsOf(_curILI);
473  foreach (Ti succ in succs)
474  {
475  int siidx = succ.Index;
476  if (_preConds[siidx].Merge(_curPostCond) ==
477  VarAssignmentSet.EMergeResult.Change)
478  {
479  q.Enqueue(siidx);
480  }
481  }
482  }

484  // Post-process to make sure each variable has at least one assignment location
485  // Variables which are not assigned in the course of the program are assumed
486  // to possess a "virtual assignment" location at IL index -1.
487  for (int i = 0; i < _numLocals; i++)
488  {
489  if (_storePoints[i].Count == 0)
490  _storePoints[i].Add(-1);

494  private void DetectEliminableAccesses()
495  {
496  _writeToRead = new Dictionary<int, int>();
497  _eliminableLocals = new HashSet<int>();
498  _writtenAndNeverRead = new HashSet<int>();
499 
500  for (int localIndex = 0; localIndex < _numLocals; localIndex++)
501  {
502  if (IsEliminationInhibited(localIndex))
503  continue;
504 
505  HashSet<int> storePoints = _storePoints[localIndex];
506  int numEliminatedAccesses = 0;
507  foreach (int storePoint in storePoints)
508  {
509  if (storePoint < 0)
510  continue;
511 
512  var storeBB = Code.GetBasicBlockContaining(storePoint);
513  HashSet<int> uses = _varUses.Get(storePoint);
514 
515  // Assignment never used?
516  if (DetectWriteAndNeverRead && uses.Count == 0)
517  {
518  _writtenAndNeverRead.Add(storePoint);
519  numEliminatedAccesses++;
520  }
521  // Used exactly once?
522  else if (DetectReadAfterWrite && uses.Count == 1)
523  {
524  int readPoint = uses.Single();
525  var readBB = Code.GetBasicBlockContaining(readPoint);
526 
527  // If variable is read within a different basic block than is
528  // is stored, do not optimize.
529  if (DoNotOptimizeAcrossBasicBlocks && !readBB.Equals(storeBB))
530  continue;
531 
532  VarAssignmentSet preCond = _preConds[readPoint];
533  if (preCond.HasSideEffects(localIndex))
534  continue;
535  IEnumerable<int> readPointStores = preCond.GetStorePoints(localIndex);
536  // Variable definition must be unambiguous
537  if (readPointStores.Count() != 1)
538  continue;
539 
540  Debug.Assert(readPointStores.Single() == storePoint);
541 
542  //_readAfterWritePairs.Add(new ReadAfterWritePair(storePoint, readPoint));
543  _writeToRead[storePoint] = readPoint;
544  numEliminatedAccesses++;
545  }
546  }
547 
548  if (numEliminatedAccesses == storePoints.Count)
549  {
550  // All variable accesses could be eliminated
551  // ==> local variable is completely superfluous
552  _eliminableLocals.Add(localIndex);
553  }
554  }
555  }

557  private void RenameLocals()
558  {
559  HashBasedPropMap<int, int> indexMap = new HashBasedPropMap<int, int>();
560  int nextIndex = 0;
561 
562  for (int localIndex = 0; localIndex < _numLocals; localIndex++)
563  {
564  HashSet<int> storePoints = _storePoints[localIndex];
565  foreach (int storePoint in storePoints)
566  {
567  if (!indexMap.ContainsKey(storePoint))
568  {
569  indexMap[storePoint] = nextIndex++;
570  }
571  HashSet<int> uses = _varUses.Get(storePoint);
572  foreach (int usePoint in uses)
573  {
574  if (!indexMap.ContainsKey(usePoint))
575  {
576  indexMap[usePoint] = nextIndex++;
577  }
578  }
579  }
580  }
581 
582  ISetAdapter<int> a = new SetAdapter(indexMap);
583  IEnumerable<int> orderedKeys = indexMap.Keys.OrderBy(key => indexMap[key]);
584  UnionFind<int> uf = new UnionFind<int>(a, orderedKeys.ToList());
585 
586  for (int localIndex = 0; localIndex < _numLocals; localIndex++)
587  {
588  HashSet<int> storePoints = _storePoints[localIndex];
589  foreach (int storePoint in storePoints)
590  {
591  HashSet<int> uses = _varUses.Get(storePoint);
592  foreach (int usePoint in uses)
593  {
594  uf.Union(storePoint, usePoint);
595  }
596  }
597  }
598 
599  _renamings = new Dictionary<Tuple<int, int>, int>();
600  for (int localIndex = 0; localIndex < _numLocals; localIndex++)
601  {
602  HashSet<int> storePoints = _storePoints[localIndex];
603  nextIndex = 0;
604  foreach (int storePoint in storePoints)
605  {
606  int repr = uf.Find(storePoint);
607  int index;
608  if (!_renamings.TryGetValue(Tuple.Create(localIndex, repr), out index))
609  {
610  index = nextIndex++;
611  _renamings[Tuple.Create(localIndex, repr)] = index;
612  }
613  _renamings[Tuple.Create(localIndex, storePoint)] = index;
614  HashSet<int> uses = _varUses.Get(storePoint);
615  foreach (int usePoint in uses)
616  {
617  _renamings[Tuple.Create(localIndex, usePoint)] = index;
618  }
619  }
620  }
