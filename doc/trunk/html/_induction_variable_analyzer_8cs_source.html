321  public void AcceptLoopBlock(LoopBlock stmt)
322  {
323  if (stmt.Initializer != null &&
324  stmt.CounterStart != null &&
325  stmt.CounterStop != null &&
326  stmt.CounterStep != null)
327  {
328  if (/*stmt.CounterStart.IsConst(_ecr) &&
329  stmt.CounterStop.IsConst(_ecr) &&*/
330  stmt.CounterStep.IsConst(_ecr))
331  {
332  try
333  {
334  Expression minExpr, maxExpr;
335  switch (stmt.CounterDirection)
336  {
337  case LoopBlock.ECounterDirection.Increment:
338  minExpr = stmt.CounterStart;
339  maxExpr = stmt.CounterStop + stmt.CounterStep;
340  if (stmt.CounterLimitKind == LoopBlock.ELimitKind.ExcludingStopValue)
341  maxExpr -= LiteralReference.CreateConstant((long)1);
342  break;
343 
344  case LoopBlock.ECounterDirection.Decrement:
345  maxExpr = stmt.CounterStart;
346  minExpr = stmt.CounterStop - stmt.CounterStep;
347  if (stmt.CounterLimitKind == LoopBlock.ELimitKind.ExcludingStopValue)
348  minExpr += LiteralReference.CreateConstant((long)1);
349  break;
350 
351  default:
352  throw new NotImplementedException();
353  }
354 
355  object min = minExpr.Eval(_eval);
356  object max = maxExpr.Eval(_eval);
357  IVRange minR = IVRange.ToRange(min);
358  IVRange maxR = IVRange.ToRange(max);
359  IVRange loopRange = IVRange.Max(minR, maxR);
360 
361  if (_iva._ivLoopRange.ContainsKey(stmt.CounterVariable))
362  _iva._ivLoopRange[stmt.CounterVariable] =
363  IVRange.Max(_iva._ivLoopRange[stmt.CounterVariable], loopRange);
364  else
365  _iva._ivLoopRange[stmt.CounterVariable] = loopRange;
