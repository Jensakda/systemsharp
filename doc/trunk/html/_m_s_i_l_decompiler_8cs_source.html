1513  {
1514  MSILCodeBlock altTarget = TrackBranchChain(targetBlock);
1515 
1516  if (_switchStack.Count > 0)
1517  {
1518  SwitchInfo si = _switchStack.Peek();
1519  int index = Array.IndexOf(si.SwitchTargets, targetBlock);
1520  if (index >= 0)
1521  {
1522  if (si.Condition == _curBB)
1523  {
1524  MSILCodeBlock cur = _curBB;
1525  DeclareBlock(targetBlock);
1526  _curBB = cur;
1527  return EBranchImpl.Nested;
1528  }
1529  else
1530  {
1531  GotoCase(si.Case, index);
1532  return EBranchImpl.ControlStatement;
1533  }
1534  }
1535  }
1536 
1537  var loopHeaders = _loopStack.Where(li => li.Header == targetBlock);
1538  if (loopHeaders.Any())
1539  {
1540  LoopInfo li = loopHeaders.First();
1541  Continue(li.Loop);
1542  return EBranchImpl.ControlStatement;
1543  }
1544 
1545  foreach (BreakInfo bi in _breakStack)
1546  {
1547  if (TrackBranchChain(bi.BreakNext) == altTarget)
1548  {

1552  Break((LoopBlock)bi.Stmt);
1553  }
1554  else
1555  {
1556  Break((CaseStatement)bi.Stmt);
1557  }
1558  return EBranchImpl.ControlStatement;
1559  }
1560  }
1561

1563  _nextStack.Peek() == targetBlock)
1564  {

1566  return EBranchImpl.Empty;
1567  }

1570  {

1572  // => implementation can be nested.
1573  MSILCodeBlock cur = _curBB;
1574  DeclareBlock(targetBlock);
