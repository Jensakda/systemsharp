121  private Statement GetAtomicSuccessor(Statement stmt)
122  {
123  do
124  {
125  Statement succ = stmt.Successor;
126  if (succ == null)
127  return null;
128  Statement asucc = succ.GetInnermostAtomicStatement();
129  if (asucc != null)
130  return asucc;
131  stmt = succ;
132  } while (true);
133  }
