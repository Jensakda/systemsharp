404  public static string ValueOf(Time value)
405  {
406  // VHDL doesn't support decimal durations, i.e. 5.123 us is syntactically wrong.
407  // Instead, we have to write it as 5123 ns. The following loop refines the time
408  // unit step-by-step from from sec -> ms -> us -> ns -> ps -> fs until we can safely
409  // truncate the fractional digits.
410  while (value.Value != 0.0 &&
411  Math.Abs(Math.Floor(value.Value) - value.Value) / value.Value > 5e-16)
412  {
413  switch (value.Unit)
414  {
415  case ETimeUnit.fs:
416  break;
417  case ETimeUnit.ms:
418  value = new Time(value.ScaleTo(ETimeUnit.us), ETimeUnit.us);
419  continue;
420  case ETimeUnit.ns:
421  value = new Time(value.ScaleTo(ETimeUnit.ps), ETimeUnit.ps);
422  continue;
423  case ETimeUnit.ps:
424  value = new Time(value.ScaleTo(ETimeUnit.fs), ETimeUnit.fs);
425  continue;
426  case ETimeUnit.sec:
427  value = new Time(value.ScaleTo(ETimeUnit.ms), ETimeUnit.ms);
428  continue;
429  case ETimeUnit.us:
430  value = new Time(value.ScaleTo(ETimeUnit.ns), ETimeUnit.ns);
431  continue;
432  default:
433  throw new NotImplementedException();
434  }
435  break;
436  }
437  return ((long)value.Value).ToString() + value.Unit.ToString();
438  }
