323  public void Save()
324  {
325  PostInitializeProperties();
326 
327  string path = MakeFullPath(ProjectName + ".xise");
328  XmlTextWriter wr = new XmlTextWriter(path, Encoding.UTF8);
329  wr.Formatting = Formatting.Indented;
330  wr.WriteStartDocument(false);
331  //wr.WriteStartElement("project", "http://www.xilinx.com/XMLSchema");
332  wr.WriteStartElement("project");
333  wr.WriteAttributeString("xmlns", "http://www.xilinx.com/XMLSchema");
334  wr.WriteAttributeString("xmlns:xil_pn", "http://www.xilinx.com/XMLSchema");
335  wr.WriteStartElement("header");
336  wr.WriteEndElement();
337 
338  wr.WriteStartElement("version");
339 
340  wr.WriteAttributeString("xil_pn:ise_version", PropEnum.ToString(ISEVersion, EPropAssoc.ISE));
341  wr.WriteAttributeString("xil_pn:schema_version", "2");
342  wr.WriteEndElement();
343 
344  var libraries = new HashSet<string>();
345  string ucf = null;
346 
347  wr.WriteStartElement("files");
348  foreach (string file in _projectFiles)
349  {
350  string type = GetFileType(file);
351  wr.WriteStartElement("file");
352  wr.WriteAttributeString("xil_pn:name", MakeUNIXPath(file));
353  wr.WriteAttributeString("xil_pn:type", type);
354  EComponentPurpose purpose = LookupAttribute<EComponentPurpose>(file);
355  if (purpose == EComponentPurpose.SimulationAndSynthesis ||
356  purpose == EComponentPurpose.SimulationOnly)
357  {
358  wr.WriteStartElement("association");
359  wr.WriteAttributeString("xil_pn:name", "BehavioralSimulation");
360  wr.WriteEndElement();
361  }
362  if (purpose == EComponentPurpose.SimulationAndSynthesis ||
363  purpose == EComponentPurpose.SynthesisOnly)
364  {
365  wr.WriteStartElement("association");
366  wr.WriteAttributeString("xil_pn:name", "Implementation");
367  wr.WriteEndElement();
368  }
369  var library = LookupAttribute<LibraryAttribute>(file);
370  if (library != null)
371  {
372  wr.WriteStartElement("library");
373  wr.WriteAttributeString("xil_pn:name", library.Name);
374  libraries.Add(library.Name);
375  wr.WriteEndElement();
376  }
377  wr.WriteEndElement();
378  }
379  wr.WriteEndElement();
380 
381  wr.WriteStartElement("properties");
382  IList<PropDesc> allProps = PropEnum.EnumProps(typeof(EXilinxProjectProperties));
383  foreach (PropDesc pdesc in allProps)
384  {
385  string propID;
386  if (!pdesc.IDs.TryGetValue(EPropAssoc.ISE, out propID))
387  continue;
388  object propVal;
389  //if (!Properties.TryGetValue(propID, out propVal))
390  if (!Properties.TryGetValue((EXilinxProjectProperties)pdesc.EnumValue, out propVal))
391  propVal = pdesc.DefaultValue;
392  bool isDefault = propVal.Equals(pdesc.DefaultValue);
393  string propValStr = PropEnum.ToString(propVal, EPropAssoc.ISE);
394  wr.WriteStartElement("property");
395  wr.WriteAttributeString("xil_pn:name", propID);
396  wr.WriteAttributeString("xil_pn:value", propValStr);
397  wr.WriteAttributeString("xil_pn:valueState", isDefault ? "default" : "non-default");
398  wr.WriteEndElement();
399  }
400  wr.WriteEndElement();
401 
402  wr.WriteStartElement("bindings");
403  if (ucf != null && TopLevelComponent != null)
404  {
405  var gi = TopLevelComponent.QueryAttribute<VHDLGenInfo>();
406  wr.WriteStartElement("binding");
407  wr.WriteAttributeString("xil_pn:location", "/" + gi.EntityName);
408  wr.WriteAttributeString("xil_pn:name", ucf);
409  wr.WriteEndElement();
410  }
411  wr.WriteEndElement();
412 
413  wr.WriteStartElement("libraries");
414  foreach (string lib in libraries)
415  {
416  wr.WriteStartElement("library");
417  wr.WriteAttributeString("xil_pn:name", lib);
418  wr.WriteEndElement();
419  }
420  wr.WriteEndElement();
421 
422  wr.WriteStartElement("autoManagedFiles");
423  wr.WriteEndElement();
424 
425  wr.WriteEndElement();
426  wr.Close();
427 
428  if (TwinProject != null)
429  TwinProject.Save();
