417  private void AnalyzeBranches()
418  {
419  MarkAsBranchTarget(Instructions[_entryPoint]);
420  Ti[] term = new Ti[] { _marshal };
421  MarkAsBranch(_marshal);
422  MarkAsBranchTarget(_marshal);
423  foreach (Ti ili in Instructions)
424  {
425  if (_exitPoints != null && _exitPoints.Contains(ili.Index))
426  {
427  _successors[ili.Index] = term;
428  MarkAsBranch(ili);
429  }
430  else if (ili.Index == _marshal.Index)
431  {
432  _successors[ili.Index] = new Ti[0];
433  }
434  else
435  {
436  IEnumerable<int> targets;
437  EBranchBehavior bb = InstructionInfo.IsBranch(ili, out targets);
438  bool includeNext = false;
439  bool isBranch = false;
440  bool isTerm = false;
441  switch (bb)
442  {
443  case EBranchBehavior.NoBranch:
444  includeNext = true;
445  isBranch = false;
446  isTerm = false;
447  break;
448 
449  case EBranchBehavior.UBranch:
450  includeNext = false;
451  isBranch = true;
452  isTerm = false;
453  break;
454 
455  case EBranchBehavior.CBranch:
456  includeNext = true;
457  isBranch = true;
458  isTerm = false;
459  break;
460 
461  case EBranchBehavior.Return:
462  case EBranchBehavior.Throw:
463  includeNext = false;
464  isBranch = true;
465  isTerm = true;
466  break;
467 
468  case EBranchBehavior.Switch:
469  includeNext = true;
470  isBranch = true;
471  isTerm = false;
472  break;
473 
474  default:
475  throw new NotImplementedException();
476  }
477 
478  if (isBranch)
479  MarkAsBranch(ili);
480 
481  foreach (int target in targets)
482  MarkAsBranchTarget(Instructions[target]);
483 
484  if (isTerm && ili.Index < Instructions.Count - 1)
485  {

487  }
488  else if (includeNext)

492  succs.AddRange(targets.Select(t => Instructions[t]));
493  _successors[ili.Index] = succs.ToArray();

495  else
