165  private EResultCategory GetResultCategory(TypeDescriptor type)
166  {
167  Type ltype = type.CILType;
168  if (ltype.Equals(typeof(sbyte)) || ltype.Equals(typeof(byte)) || ltype.Equals(typeof(short)) || ltype.Equals(typeof(ushort)) ||
169  ltype.Equals(typeof(int)) || ltype.Equals(typeof(uint)) || ltype.Equals(typeof(long)) || ltype.Equals(typeof(bool)) ||
170  ltype.Equals(typeof(char)))
171  return EResultCategory.SignedIntegral;
172  else if (ltype.Equals(typeof(ulong)))
173  return EResultCategory.UnsignedIntegral;
174  else if (ltype.Equals(typeof(float)) || ltype.Equals(typeof(double)))
175  return EResultCategory.FloatingPoint;
176  else
177  return EResultCategory.Uncountable;

180  private bool TryCallIntrinsicOp(string name, out object result, params object[] args)
181  {
182  Type[] argTypes = new Type[args.Length];
183  for (int i = 0; i < args.Length; i++)
184  {
185  object arg = args[i];
186  argTypes[i] = arg == null ? typeof(object) : arg.GetType();
187  }
188  for (int i = 0; i < args.Length; i++)
189  {
190  MethodInfo mi = argTypes[i].GetMethod(name, argTypes);
191  if (mi != null)
192  {
193  result = mi.Invoke(null, args);
194  return true;
195  }
196  }
197  result = null;
198  return false;
199  }
