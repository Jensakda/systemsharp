429  public void SaveToXSTScript(string path)
430  {
431  var sw = new StreamWriter(path);
432  if (TempDir != null)
433  sw.WriteLine("set -tmpdir \"" + TempDir + "\"");
434  if (XstHdpDir != null)
435  sw.WriteLine("set -xsthdpdir \"" + XstHdpDir + "\"");
436  sw.WriteLine("run");
437  foreach (var prop in typeof(XSTFlow).GetProperties())
438  {
439  if (!Attribute.IsDefined(prop, typeof(FlowProp)))
440  continue;
441  object value = prop.GetValue(this, new object[0]);
442  if (value == null)
443  continue;
444  var flowProp = (FlowProp)Attribute.GetCustomAttribute(prop, typeof(FlowProp));
445  sw.Write("-" + flowProp.ID);
446  sw.Write(" ");
447  if (value.GetType().IsEnum)
448  {
449  sw.WriteLine(PropEnum.ToString(value, EPropAssoc.XST));
450  }
451  else
452  {
453  sw.WriteLine(value.ToString());
454  }
455  }
456  sw.WriteLine();
457  sw.Close();
458  }
