264  class PropEnum
265  {
266  public static IList<PropDesc> EnumProps(Type type)
267  {
268  FieldInfo[] fields = type.GetFields();
269  List<PropDesc> result = new List<PropDesc>();
270  foreach (FieldInfo field in fields)
271  {
272  if (!field.IsStatic)
273  continue;
274  object value = field.GetValue(null);
275  object[] valueTypes = field.GetCustomAttributes(typeof(PropValueType), false);
276  Type valueType = null;
277  object defaultValue = null;
278  foreach (PropValueType pvt in valueTypes)
279  {
280  valueType = pvt.Type;
281  defaultValue = pvt.DefaultValue;
282  }
283  PropDesc pd = new PropDesc(value, valueType, defaultValue);
284  object[] propIDs = field.GetCustomAttributes(typeof(PropID), false);
285  foreach (PropID pid in propIDs)
286  {
287  pd.IDs[pid.Assoc] = pid.ID;
288  }
289  result.Add(pd);
290  }
291  return result;
292  }

294  public static string ToString(object value, EPropAssoc assoc)
295  {
296  if (value is bool)
297  {
298  bool bvalue = (bool)value;
299  return bvalue ? "true" : "false";
300  }
301  Type type = value.GetType();
302  IList<PropDesc> allFields = EnumProps(type);
303  PropDesc pd = (from PropDesc pdi in allFields
304  where pdi.EnumValue.Equals(value)
305  select pdi).SingleOrDefault();
306  string result;
307  if (pd == null || !pd.IDs.TryGetValue(assoc, out result))
308  {
309  if (value is double)
310  {
311  double dbl = (double)value;
312  result = dbl.ToString(CultureInfo.InvariantCulture);
313  }
314  else
315  {
316  result = value.ToString();
317  }
318  }
319  return result;
320  }
