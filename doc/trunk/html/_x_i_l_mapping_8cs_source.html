264

266  if (_xilMappings.TryGetValue(xilsi, out mapping))
267  {

306  public IXILMapping TryBind(XIL3Instr instr, long cstep, TypeDescriptor[] operandTypes, TypeDescriptor[] resultTypes)
307  {
308  ReservationTable rtbl;
309  var mappers = _xmm.LookupMappers(instr.Command);
310  var viableMappings = new List<IXILMapping>();
311  foreach (IXILMapper mapper in mappers)
312  {
313  var tas = _taBindLookup.Get(mapper);
314  foreach (var ta in tas)
315  {
316  var mappings = mapper.TryMap(ta, instr.Command, operandTypes, resultTypes);
317  rtbl = _resTables[ta];
318  foreach (var mapping in mappings)
319  {
320  if (mapping.InitiationInterval > 0)
321  {
322  if (!rtbl.IsReserved(cstep, cstep + mapping.InitiationInterval - 1))
323  viableMappings.Add(mapping);
324  else if (mapping.ResourceKind == EMappingKind.ExclusiveResource)
325  return null;
326  }
327  else
328  {
329  viableMappings.Add(mapping);
330  }
331  }
332  }
333  }
334  bool allocateNew = true;
335  IXILMapping bestMapping = null;
336  int lat = 0;
337  if (viableMappings.Count > 0)
338  {
339  lat = viableMappings.First().Latency;
340  if (!viableMappings.All(m => m.Latency == lat))
341  throw new XILSchedulingFailedException("Mappings with different latencies exist for " + instr);
342 
343  if (viableMappings.All(m => m.ResourceKind == EMappingKind.ExclusiveResource))
344  {
345  allocateNew = false;
346  bestMapping = viableMappings.First();
347  }
348  else if (viableMappings.All(m => m.ResourceKind == EMappingKind.LightweightResource))
349  {
350  allocateNew = true;
351  }
352  else
353  {
354  allocateNew = Policy.SelectBestMapping(instr, cstep, viableMappings, out bestMapping) ==
355  EAllocationDecision.AllocateNew;
356  }
357  }
358  if (allocateNew)
359  {
360  foreach (IXILMapper mapper in mappers)
361  {
362  bestMapping = mapper.TryAllocate(_host, instr.Command, operandTypes, resultTypes, _targetProject);
363  if (bestMapping != null)
364  {
365  if (viableMappings.Count > 0 && bestMapping.Latency != lat)
366  throw new XILSchedulingFailedException("Newly allocated mapping for " + instr + " has different latency.");
367 
368  if (_onAllocation != null)
369  _onAllocation(bestMapping);
370  _taBindLookup.Add(mapper, bestMapping.TASite);
371  //bestMapping.TASite.Establish(binder);
372  break;
373  }
374  }
375  }
376  if (bestMapping == null)
377  return null;
378  rtbl = _resTables[bestMapping.TASite];
379  bool ok = rtbl.TryReserve(cstep, cstep + bestMapping.InitiationInterval - 1, instr);
380  Debug.Assert(ok);
381  return bestMapping;
382  }
