201  public long Schedule<T>(ISchedulingAdapter<T> a, IList<T> nodes, IList<T> startNodes,
202  SchedulingConstraints constraints)
203  {
204  CheckInput(a, nodes);
205 
206  long startTime = constraints.StartTime;
207 
208  foreach (T x in nodes)
209  {
210  a.CStep[x] = long.MinValue;
211  }
212 
213  var pq = new PriorityQueue<NodeSet<T>>()
214  {
215  Resolve = NodeSet<T>.Merge
216  };
217  foreach (T x in startNodes)
218  pq.Enqueue(startTime, NodeSet<T>.From(x));
219  long endTime = startTime + 1;
220  while (!pq.IsEmpty)
221  {
222  var cur = pq.Dequeue();
223  long curTime = cur.Key;
224  var curSet = cur.Value;
225  foreach (T x in curSet.Nodes)
226  {
227  bool ready = true;
228  long reqTime = curTime;
229  foreach (var dw in a.Preds[x])
230  {
231  T w = dw.Task;
232  if (a.CStep[w] == long.MinValue)
233  {
234  // at least one predecessor is not yet scheduled
235  // -> we cannot tell whether it is ok to schedule current task.
236  ready = false;
237  reqTime = long.MinValue;
238  break;
239  }
240  else if (a.CStep[w] + dw.MinDelay > curTime)
241  {
242  // at least one predecessor did not yet complete.
243  ready = false;
244  if (reqTime > long.MinValue)
245  reqTime = Math.Max(reqTime, a.CStep[w] + dw.MinDelay);
246  }
247  }
248  if (ready)
249  {
250  // Check for deadline violations in second pass
251  foreach (var dw in a.Preds[x])
252  {
253  T w = dw.Task;
254  if (a.CStep[w] + dw.MaxDelay < curTime)
255  {
256  // deadline exceeded
257  throw new NotSchedulableException();
258  }
259  }
260 
261  if (a.CStep[x] == long.MinValue)
262  {
263  long preHint, postHint;
264  if (!ConstrainedResources || a.TryPin(x, curTime, out preHint, out postHint))
265  {
266  a.CStep[x] = curTime;
267  long lat = a.Latency[x];
268  long nextTime = curTime + lat;
269  if (lat > 0)
270  endTime = Math.Max(endTime, nextTime);
271  else
272  endTime = Math.Max(endTime, nextTime + 1);
273 
274  // enqueue successor tasks
275  foreach (var dy in a.Succs[x])
276  {
277  T y = dy.Task;
278  pq.Enqueue(curTime + dy.MinDelay, NodeSet<T>.From(y));
279  }
280  }
281  else
282  {
283  pq.Enqueue(postHint, NodeSet<T>.From(x));
284  }
285  }
286  }
287  else if (reqTime > long.MinValue)
288  {
289  pq.Enqueue(reqTime, NodeSet<T>.From(x));
290  }
291  }
292  }
293  foreach (T x in nodes)
294  {
295  if (a.CStep[x] == long.MinValue)
296  throw new NotSchedulableException();
297  }
298  return endTime;
299  }

358  public long Schedule<T>(ISchedulingAdapter<T> a, IList<T> nodes, IList<T> end, SchedulingConstraints constraints)
359  {
360  ASAPScheduler.CheckInput(a, nodes);
361 
362  long endTime = constraints.EndTime;
363 
364  foreach (T x in nodes)
365  {
366  a.CStep[x] = long.MaxValue;
367  }
368 
369  var pq = new PriorityQueue<NodeSet<T>>()
370  {
371  Resolve = NodeSet<T>.Merge
372  };
373  foreach (T x in end)
374  pq.Enqueue(-endTime, NodeSet<T>.From(x));
375  long startTime = endTime - 1;
376  while (!pq.IsEmpty)
377  {
378  var cur = pq.Dequeue();
379  long curTime = -cur.Key;
380  long reqTime = long.MaxValue;
381  var curSet = cur.Value;
382  var order = PriorizeNodes(curSet.Nodes);
383  foreach (T x in order)
384  {
385  bool ready = true;
386  foreach (var dy in a.Succs[x])
387  {
388  T y = dy.Task;
389  if (a.CStep[y] == long.MaxValue)
390  {
391  // at least one successor is not yet scheduled
392  // -> task not ready
393  ready = false;
394  reqTime = long.MaxValue;
395  break;
396  }
397  else if (a.CStep[y] < curTime - a.Latency[x] + dy.MinDelay)
398  {
399  // at least one successor starts before current task could
400  // complete -> task not ready
401  ready = false;
402  if (reqTime < long.MaxValue)
403  reqTime = Math.Min(reqTime, a.CStep[y] + a.Latency[x] - dy.MinDelay);
404  }
405  }
406  if (ready)
407  {
408  // Check for deadline violations in second pass
409  foreach (var dy in a.Succs[x])
410  {
411  T y = dy.Task;
412  if (a.CStep[y] > curTime - a.Latency[x] + dy.MaxDelay)
413  {
414  // deadline exceeded
415  throw new NotSchedulableException();
416  }
417  }
418 
419  long lat = a.Latency[x];
420  long execTime = curTime - lat;
421  // If some operation is combinatorial (latency == 0) and is scheduled as
422  // last instruction, it must be moved one step back to fit inside the schedule's
423  // time frame.
424  if (execTime == endTime)
425  --execTime;
426  long preHint, postHint;
427  if (!ConstrainedResources || a.TryPin(x, execTime, out preHint, out postHint))
428  {
429  a.CStep[x] = execTime;
430  long nextTime = execTime;
431  startTime = Math.Min(startTime, execTime);
432  // requeue predecessors
433  foreach (var dw in a.Preds[x])
434  {
435  T w = dw.Task;
436  pq.Enqueue(-(execTime - dw.MinDelay + a.Latency[w]), NodeSet<T>.From(w));
437  }
438  }
439  else
440  {
441  if (preHint < 0)
442  throw new NotSchedulableException();
443 
444  pq.Enqueue(-(preHint+lat), NodeSet<T>.From(x));
445  }
446  }
447  else if (reqTime < long.MaxValue)
448  {
449  pq.Enqueue(-reqTime, NodeSet<T>.From(x));
450  }
451  }
452  }
453  foreach (T x in nodes)
454  {
455  if (a.CStep[x] == long.MaxValue)
456  throw new NotSchedulableException();
457  }
458  return startTime;
459  }

496  public virtual void Schedule<T>(ControlFlowGraph<T> cfg, SchedulingConstraints constraints, ISchedulingAdapter<T> scha) where T : Analysis.IInstruction
497  {

502  if (bb.IsExitBlock)
503  break;
504 
505  constraints.EndTime = cur;
506  Schedule(bb.Range, scha, constraints);
507  endTimes.Enqueue(constraints.EndTime - constraints.StartTime);
508  }
509  scha.ClearSchedule();
510  constraints.EndTime = 0;
511  foreach (var bb in cfg.BasicBlocks)
512  {
513  if (bb.IsExitBlock)
514  break;
515
