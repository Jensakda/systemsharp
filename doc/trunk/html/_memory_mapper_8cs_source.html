152  private void EmitIndexComputation(Array array)
153  {
154  MemoryMappedStorage mms = GetDataLayout(array);
155  ArrayMemoryLayout layout = mms.Layout as ArrayMemoryLayout;
156  if (layout.ElementsPerWord > 1)
157  throw new NotImplementedException("Multiple elements per word not yet implemented");
158  MemoryRegion region = mms.Region;
159  IMarshalInfo minfo = region.MarshalInfo;
160 
161  int shiftWidth = MathExt.CeilLog2(mms.Region.AddressWidth);
162  TypeDescriptor indexType = TypeDescriptor.GetTypeOf(mms.BaseAddress);
163 
164  for (int i = 0; i < layout.Strides.Length; i++)
165  {
166  TypeDescriptor orgIndexType;
167  if (i > 0)
168  {
169  orgIndexType = TypeStack.Skip(1).First();
170  Emit(_iset.Swap().CreateStk(2, orgIndexType, indexType, indexType, orgIndexType));
171  }
172  else
173  {
174  orgIndexType = TypeStack.Peek();
175  }
176 
177  ulong stride = layout.Strides[layout.Strides.Length - i - 1];
178 
179  if (MathExt.IsPow2(stride))
180  {
181  Emit(_iset.Convert().CreateStk(1, orgIndexType, indexType));
182  int ishift = MathExt.CeilLog2(stride);
183  if (ishift > 0)
184  {
185  Unsigned shift = Unsigned.FromULong((ulong)ishift, shiftWidth);
186  TypeDescriptor shiftType = TypeDescriptor.GetTypeOf(shift);
187  Emit(_iset.LdConst(shift).CreateStk(0, shiftType));
188  Emit(_iset.LShift().CreateStk(2, indexType, shiftType, indexType));
189  }
190  }
191  else
192  {
193  throw new NotImplementedException("Stride ain't a power of 2");
194  }
195 
196  if (i > 0)
197  {
198  Emit(_iset.Or().CreateStk(2, indexType, indexType, indexType));
199  }
200  }
201 
202  if (mms.BaseAddress.ULongValue != 0)
203  {
204  Emit(_iset.LdConst(mms.BaseAddress).CreateStk(0, indexType));
205  if (minfo.UseStrongPow2Alignment)
206  Emit(_iset.Or().CreateStk(2, indexType, indexType, indexType));
207  else
208  Emit(_iset.Add().CreateStk(2, indexType, indexType, indexType));
209  }
210  }

623  Emit(_iset.Convert().CreateStk(1, concatType, rawWordType));
624  }

633  Emit(_iset.RShift().CreateStk(1, concatType, concatType));
634  }

636  }

639  protected override void PostProcess()
640  {
641  base.PostProcess();
642  var bbs = XILSInstructionInfo.GetBasicBlocks(OutInstructions);
643  var mmap = new Dictionary<MemoryRegion, int>();
644  foreach (var bb in bbs)
645  {
646  foreach (var xilsi in bb.ToArray())
647  {
648  MemoryRegion rgn = null;
649  switch (xilsi.Name)
650  {
651  case InstructionCodes.RdMem:
652  case InstructionCodes.WrMem:
653  rgn = (MemoryRegion)xilsi.StaticOperand;
654  break;
655 
656  case InstructionCodes.RdMemFix:
657  case InstructionCodes.WrMemFix:
658  rgn = ((Tuple<MemoryRegion, Unsigned>)xilsi.StaticOperand).Item1;
659  break;
660  }
661  if (rgn != null)
662  {
663  int pred;
664  if (mmap.TryGetValue(rgn, out pred))
665  {
666  var preds = xilsi.Preds.Union(
667  new InstructionDependency[] {
668  new OrderDependency(pred, OrderDependency.EKind.BeginAfter) }).ToArray();
669  var newXilsi = xilsi.Command.CreateStk(preds, xilsi.OperandTypes, xilsi.ResultTypes);
670  newXilsi.Index = xilsi.Index;
671  OutInstructions[xilsi.Index] = newXilsi;
672  }
673  mmap[rgn] = xilsi.Index;
674  }
675  }
676  mmap.Clear();
677  }
