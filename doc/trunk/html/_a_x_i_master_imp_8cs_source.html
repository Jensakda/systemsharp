699  public void Write(string textToAppend)
703  {
704  if (string.IsNullOrEmpty(textToAppend))
705  {
706  return;
707  }
708  // If we're starting off, or if the previous text ended with a newline,
709  // we have to append the current indent first.
710  if (((this.GenerationEnvironment.Length == 0)
711  || this.endsWithNewline))
712  {
713  this.GenerationEnvironment.Append(this.currentIndentField);
714  this.endsWithNewline = false;
715  }
716  // Check if the current text ends with a newline
717  if (textToAppend.EndsWith(global::System.Environment.NewLine, global::System.StringComparison.CurrentCulture))
718  {
719  this.endsWithNewline = true;
720  }
721  // This is an optimization. If the current indent is "", then we don't have to do any
722  // of the more complex stuff further down.
723  if ((this.currentIndentField.Length == 0))
724  {
725  this.GenerationEnvironment.Append(textToAppend);
726  return;
727  }
728  // Everywhere there is a newline in the text, add an indent after it
729  textToAppend = textToAppend.Replace(global::System.Environment.NewLine, (global::System.Environment.NewLine + this.currentIndentField));

731  // because the appropriate indent will be added when the next time Write() is called
732  if (this.endsWithNewline)

734  this.GenerationEnvironment.Append(textToAppend, 0, (textToAppend.Length - this.currentIndentField.Length));
735  }
736  else
737  {

739  }
