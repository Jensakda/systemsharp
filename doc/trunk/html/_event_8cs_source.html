158  public override bool Rewrite(Meta.CodeDescriptor decompilee, SysDOM.Expression waitObject, Analysis.IDecompiler stack, SysDOM.IFunctionBuilder builder)
159  {
160  var evt = waitObject.ResultType.GetSampleInstance();
161  var sevent = evt as Event;
162  if (sevent == null)
163  {
164  // This workaround is for the following situation:
165  // Signal s;
166  // await s;
167  // This will actually await the "changed event" which is associated with the signal. However, decompilation
168  // will treat the signal instance as the awaited object and pass a Signal instance instead of the event.
169  // The code will try to restore the Event instance from the Awaiter.
170  Awaiter awaiter = null;
171  try
172  {
173  awaiter = ((object)AwaitableExtensionMethods.GetAwaiter((dynamic)evt)) as Awaiter;
174  }
175  catch
176  {
177  }

179  throw new InvalidOperationException("Unable to resolve awaited MultiEvent");

181  }

184  var signalRef = SignalRef.Create(signal, SignalRef.EReferencedProperty.Instance);
