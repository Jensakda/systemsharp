1288  {

1290  if (!_converters.TryGetValue(SType, out map))
1291  return null;

1295  {
1296  MethodInfo convm = convms.Where(mi => mi.GetParameters().Length == args.Length).SingleOrDefault();
1297  if (convm != null)
1298  return (string)convm.Invoke(null, args.Cast<object>().ToArray());
1299  }
1300  visited.Push(map);
1301  string full = null;
1302  foreach (var kvp in map)
1303  {
1304  Type ttype = kvp.Key;
1305  if (!_converters.TryGetValue(ttype, out tmap))
1306  continue;
1307  if (visited.Contains(tmap))
1308  continue;
1309 
1310  convms = kvp.Value;
1311  MethodInfo convm = convms.Where(mi => mi.GetParameters().Length == 1).SingleOrDefault();
1312  if (convm == null)
1313  continue;
1314  string temp = (string)convm.Invoke(null, new object[] { args[0] });
1315  string[] targs = (string[])args.Clone();
1316  targs[0] = temp;
1317  full = ConvertDFS(ttype, TType, visited, targs);
1318  if (full != null)
1319  {
1320  break;
1321  }
1322  }
1323  visited.Pop();

1327  public static string Convert(Type SType, TypeDescriptor TTypeD, IList<Type> interTypes, params string[] args)
1328  {
1329  var TType = TTypeD.CILType;
1330 
1331  if (SType.Equals(TType))
1332  return args[0];
1333 
1334  TypePaths paths = _instance._typePaths[SType];
1335  if (paths.Preds[TType] == null)
1336  throw new NotSupportedException("Don't know any conversion from " + SType.Name + " to " + TType.Name);
1337 
1338  Type curT = TType;
1339  Stack<MethodInfo> convChain = new Stack<MethodInfo>();
1340  int nargs = args.Length;
1341  while (curT != SType)
1342  {
1343  interTypes.Add(curT);
1344  Type preT = paths.Preds[curT];
1345  List<MethodInfo> cands = _converters[preT][curT];
1346  MethodInfo convm = cands.Where(m => m.GetParameters().Length == nargs + 1).FirstOrDefault();
1347  if (convm != null)
1348  {
1349  convChain.Push(convm);
1350  nargs = 1;
1351  }
1352  else
1353  {
1354  convm = cands.Where(m => m.GetParameters().Length == 2).FirstOrDefault();
1355  if (convm == null)
1356  throw new NotSupportedException("Don't know any conversion from " + SType.Name + " to " + TType.Name + " given the supplied parameters");
1357  convChain.Push(convm);
1358  }
1359  curT = preT;
1360  }
1361  string[] curArgs = (string[])args.Clone();
1362  foreach (MethodInfo convm in convChain)
1363  {
1364  if (convm.GetParameters().Length == 2)
1365  {
1366  curArgs[0] = (string)convm.Invoke(null, new object[] { curArgs[0], TTypeD });
1367  }
1368  else
1369  {
1370  curArgs[0] = (string)convm.Invoke(null,
1371  curArgs.Cast<object>().Concat(Enumerable.Repeat(TTypeD, 1)).ToArray());
1372  }
1373  }
1374  return curArgs[0];
1375  }
