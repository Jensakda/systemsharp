418  public IEnumerable<ITimedFlow> GetTimedFlows()
419  {
420  var result = new List<ITimedFlow>();
421  var tempMap = new Dictionary<int, ITimedFlow>();
422  for (int cstep = 0; cstep < _graphs.Count; cstep++)
423  {
424  var g = _graphs[cstep];
425  var flows = g.ToFlow().Flows;
426  foreach (var f in flows)
427  {
428  var sf = f as SignalFlow;
429  var vf = f as ValueFlow;
430  if (sf != null)
431  {
432  ITimedFlow tf;
433  if (sf.Source.IsTemporary())
434  {
435  tf = tempMap[sf.Source.GetTemporaryIndex()];
436  var tsf = tf as TimedSignalFlow;
437  var tvf = tf as TimedValueFlow;
438  if (tsf != null)
439  tf = new TimedSignalFlow(tsf.Source, sf.Target, tf.Time, 0);
440  else
441  tf = new TimedValueFlow(tvf.Value, sf.Target, cstep);
442  }
443  else
444  {
445  tf = new TimedSignalFlow(sf.Source, sf.Target, cstep, 0);
446  }
447  if (f.Target.IsTemporary())
448  {
449  tempMap[f.Target.GetTemporaryIndex()] = tf;
450  }
451  else
452  {
453  var tsf = tf as TimedSignalFlow;
454  var tvf = tf as TimedValueFlow;
455  if (tsf != null)
456  tf = new TimedSignalFlow(tsf.Source, tf.Target, tf.Time, cstep - tf.Time);
457  else
458  tf = new TimedValueFlow(tvf.Value, tf.Target, cstep);
459  result.Add(tf);
460  }
461  }
462  else
463  {
464  var tf = new TimedValueFlow(vf.Value, f.Target, cstep);
465  if (f.Target.IsTemporary())
466  {
467  tempMap[f.Target.GetTemporaryIndex()] = tf;
468  }
469  else
470  {
471  result.Add(tf);
472  }
473  }
474  }
475  }
