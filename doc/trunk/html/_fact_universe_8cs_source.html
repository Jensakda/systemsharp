370  private void OnNewType(TypeFacts typeFacts)
371  {
372  Type type = typeFacts.TheType;
373 
374  MethodInfo[] methods = type.GetMethods(
375  BindingFlags.Static | BindingFlags.Instance |
376  BindingFlags.Public | BindingFlags.NonPublic);
377  foreach (MethodInfo method in methods)
378  {
379  if (method.GetCustomOrInjectedAttribute(typeof(IAssumeCalled)) != null ||
380  method.Name.Contains('<')) // Anonymous methods don't give a chance to attach an attribute.
381  {
382  AddMethod(method);
383  }
384  }
385  foreach (Type itype in type.GetInterfaces())
386  AddType(itype);
387  Type btype = type.BaseType;
388  while (btype != null && !btype.Equals(type))

401  private void OnNewMethod(MethodFacts mfacts)
402  {
403  mfacts.OnNewMethodCalled = true;
404  MethodBase method = mfacts.Method;
405  AddType(method.DeclaringType);
406  IOnMethodDiscovery disc = (IOnMethodDiscovery)method.GetCustomOrInjectedAttribute(typeof(IOnMethodDiscovery));
407  if (disc != null)
408  disc.OnDiscovery(method, EMethodDiscoveryStage.Processing);
409  RealizationsOf(method).AutoDo(AddMethod);
410  if (mfacts.IsDecompilable)
411  {
412  mfacts.INVA.CalledMethods.AutoDo(AddMethod);
413  mfacts.INVA.ConstructedObjects.AutoDo(AddConstructor);
414  mfacts.INVA.ReferencedTypes.AutoDo(AddType);
415  mfacts.INVA.ReadFields.AutoDo(OnFieldSeen);
416  mfacts.INVA.WrittenFields.AutoDo(OnFieldSeen);
417  mfacts.INVA.ReferencedFields.AutoDo(OnFieldSeen);
418  mfacts.INVA.ElementSources
419  .Where(src => src is AllocationSite)
420  .Select(src => (AllocationSite)src)
421  .AutoDo(src => _allocSites.Add(src));
422  mfacts.INVA.Run();
423  if (method is MethodInfo)
424  mfacts.Realizations = RealizationsOf((MethodInfo)method).ToBufferedEnumerable();
425  else
426  mfacts.Realizations = Enumerable.Repeat(method, 1);
427  }

437  public IObservable<MethodBase> RealizationsOf(MethodInfo method)
438  {
439  if (method.DeclaringType.IsInterface)
440  {
441  return KnownTypes
442  .Where(t => !(method.DeclaringType.IsGenericType && t.TheType.IsArray)
443  && !t.TheType.IsGenericParameter && !t.TheType.IsInterface
444  && method.DeclaringType.IsAssignableFrom(t.TheType))
445  .Select(t => t.TheType.GetInterfaceMap(method.DeclaringType))
446  .Select(im => Tuple.Create(im, im.InterfaceMethods.ToList().IndexOf(method)))
447  .Where(tup => tup.Item2 >= 0)
448  .Select(tup => tup.Item1.TargetMethods[tup.Item2]);
449  }
450  else if (method.IsVirtual)
451  {
452  return
453  RealizationsOf(method.DeclaringType)
454  .SelectMany(t => t.GetMethods(
455  BindingFlags.Public | BindingFlags.NonPublic |
456  BindingFlags.Instance | BindingFlags.Static)
457  .Where(m => m.GetBaseDefinition().MethodHandle == method.MethodHandle));
458  }
459  else
460  return Observable.Return(method);

494  private void PostProcess()
495  {
496  Queue<MethodFacts> q = new Queue<MethodFacts>();
497  foreach (MethodFacts mf in _knownMethods.Values)
498  {
499  foreach (ElementMutation mut in mf.Mutations)
500  {
501  foreach (ElementSource term in mut.GetMutatedTerminals())
502  {
503  if (term is IMutableSource)
504  {
505  ((IMutableSource)term).IndicateMutated();
506  }
507  }
508  foreach (ElementSource term in mut.GetSubMutatedTerminals())
509  {
510  if (term is IMutableSource)
511  {
512  ((IMutableSource)term).IndicateSubMutated();
513  }
514  }
515  }
516  if (mf.IsMutator || mf.IsUnsafe)
517  q.Enqueue(mf);
518  }
519  while (q.Count > 0)
520  {
521  MethodFacts mf = q.Dequeue();
522  foreach (MethodBase c in mf.CallingMethods)
523  {
524  MethodFacts mfc = _knownMethods[c];
525  if (!mfc.IsIndirectMutator)
526  {
527  mfc.IsIndirectMutator = true;
528  q.Enqueue(mfc);
529  }
530  }
531  }
