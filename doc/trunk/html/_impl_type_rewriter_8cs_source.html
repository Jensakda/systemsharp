87  private static TypeDescriptor GetCommonType(TypeDescriptor td1, TypeDescriptor td2)
88  {

97  Math.Max(fmt1.IntWidth, fmt2.IntWidth + 1),
98  Math.Max(fmt1.FracWidth, fmt2.FracWidth));
99  }
100  else if (IsUFix(td1) && IsSFix(td2))

108  return SFix.MakeType(
109  Math.Max(fmt1.IntWidth, fmt2.IntWidth),

111  }

113  {

116  return UFix.MakeType(
117  Math.Max(fmt1.IntWidth, fmt2.IntWidth),

119  }

185  private InstructionDependency[] Convert(InstructionDependency[] preds, TypeDescriptor from, TypeDescriptor to)
186  {
187  if (from.Equals(to))
188  {
189  // nothing to do
190  return preds;
191  }
192  else if (IsSFix(from) && IsUFix(to))
193  {
194  var fromFmt = SFix.GetFormat(from);
195  var toFmt = UFix.GetFormat(to);
196  int interIW = toFmt.IntWidth + 1;
197  int interFW = toFmt.FracWidth;
198  if (interIW != fromFmt.IntWidth ||
199  interFW != fromFmt.FracWidth)
200  {
201  var interType = SFix.MakeType(interIW, interFW);
202  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, interType));
203  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(1, interType, to));
204  }
205  else
206  {
207  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, to));
208  }
209  return new InstructionDependency[0];
210  }
211  else if (IsUFix(from) && IsSFix(to))
212  {
213  var fromFmt = UFix.GetFormat(from);
214  var toFmt = SFix.GetFormat(to);
215  int interIW = toFmt.IntWidth - 1;
216  int interFW = toFmt.FracWidth;
217  if (interIW != fromFmt.IntWidth ||
218  interFW != fromFmt.FracWidth)
219  {
220  var interType = UFix.MakeType(interIW, interFW);
221  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, interType));
222  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(1, interType, to));
223  }
224  else
225  {
226  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, to));
227  }
228  return new InstructionDependency[0];
229  }
230  else if (IsSLV(from))
231  {
232  int wfrom = TypeLowering.Instance.GetWireWidth(from);
233  int wto = TypeLowering.Instance.GetWireWidth(to);
234  if (wfrom == wto)
235  {
236  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, to));
237  }
238  else
239  {
240  var interType = StdLogicVector.MakeType(wto);
241  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, interType));
242  Convert(interType, to);
243  }
244  return new InstructionDependency[0];
245  }
246  else if (IsSLV(to))
247  {
248  int wfrom = TypeLowering.Instance.GetWireWidth(from);
249  int wto = TypeLowering.Instance.GetWireWidth(to);
250  if (wfrom == wto)
251  {
252  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, to));

257  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, from, interType));

259  }

678  {
679  var candidates = types.Where(t => t.CILType.Equals(type.CILType));
680  if (type.CILType.Equals(typeof(SFix)) ||
681  type.CILType.Equals(typeof(UFix)) ||
682  type.CILType.Equals(typeof(Signed)) ||
683  type.CILType.Equals(typeof(Unsigned)))
684  {
685  var fmt = type.GetFixFormat();
686  candidates = candidates.OrderBy(t =>
687  Math.Abs(t.GetFixFormat().IntWidth - fmt.IntWidth));
688  candidates = candidates.OrderBy(t =>
689  Math.Abs(t.GetFixFormat().FracWidth - fmt.FracWidth));
690  }
691  var result = candidates.FirstOrDefault();
692  if (result == null)
693  result = types.FirstOrDefault();
694  return result;

842  private void ProcessSelect(XILSInstr i)
843  {
844  var preds = RemapPreds(i.Preds);
845  var ot1 = i.OperandTypes[0];
846  var ot2 = i.OperandTypes[1];
847  var rt = i.ResultTypes[0];
848  var ct = GetCommonType(GetCommonType(ot1, ot2), rt);
849  if (!ot2.Equals(ct))
850  {
851  // stack: [ x y cond ]
852  Swap(preds);
853  // stack: [ x cond y ]
854  Convert(preds, ot2, ct);
855  Swap();
856  // stack: [ x y cond ]
857  preds = new InstructionDependency[0];
858  }
859  if (!ot1.Equals(ct))
860  {
861  // stack: [ x y cond ]
862  Dig(preds, 2);
863  // stack: [ y cond x ]
864  Convert(ot1, ct);
865  Dig(2);
866  // stack: [ cond x y ]
867  Dig(2);
868  // stack: [ x y cond ]
869  preds = new InstructionDependency[0];
870  }
871  var inew = DefaultInstructionSet.Instance.Select().CreateStk(preds, 3, ct, ct, i.OperandTypes[2], ct);
872  Emit(inew);
873  if (!rt.Equals(ct))
874  {
875  Convert(ct, rt);
876  }
