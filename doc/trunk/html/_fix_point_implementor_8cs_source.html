72  private InstructionDependency[] EqualizeTypes(XILSInstr i, bool makeSameSize, TypeDescriptor[] otypes)
73  {
74  var otype0 = i.OperandTypes[0];
75  var otype1 = i.OperandTypes[1];
76  FixFormat fmt0, fmt1;
77  bool flag0 = IsFix(otype0, out fmt0);
78  bool flag1 = IsFix(otype1, out fmt1);
79  if (flag0 != flag1)
80  throw new InvalidOperationException("Incompatible types");
81  var preds = RemapPreds(i.Preds);
82  if (flag0)
83  {
84  otypes[0] = otype0;
85  otypes[1] = otype1;
86  if (fmt0.IsSigned && !fmt1.IsSigned)
87  {
88  var sample = (UFix)otype1.GetSampleInstance();
89  var signed = sample.SFixValue;
90  var stype1 = TypeDescriptor.GetTypeOf(signed);
91  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, otype1, stype1));
92  otype1 = stype1;
93  fmt1 = signed.Format;
94  preds = new InstructionDependency[0];
95  otypes[1] = otype1;
96  }
97  if (!fmt0.IsSigned && fmt1.IsSigned)

100  var signed = sample.SFixValue;
101  var stype0 = TypeDescriptor.GetTypeOf(signed);
102  Emit(DefaultInstructionSet.Instance.Dig(1).CreateStk(preds, 2, otype0, otype1, otype1, otype0));
103  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(1, otype0, stype0));
104  otype0 = stype0;
105  Emit(DefaultInstructionSet.Instance.Dig(1).CreateStk(preds, 2, otype1, otype0, otype0, otype1));
106  fmt0 = signed.Format;
107  preds = new InstructionDependency[0];
108  otypes[0] = otype0;
109  }
110  if (makeSameSize)
111  {
112  int intWidth = Math.Max(fmt0.IntWidth, fmt1.IntWidth);
113  int fracWidth = Math.Max(fmt0.FracWidth, fmt1.FracWidth);
114  object rsample;
115  if (fmt0.IsSigned)
116  rsample = SFix.FromDouble(0.0, intWidth, fracWidth);
117  else
118  rsample = UFix.FromDouble(0.0, intWidth, fracWidth);
119  var rtype = TypeDescriptor.GetTypeOf(rsample);
120  if (intWidth > fmt0.IntWidth || fracWidth > fmt0.FracWidth)
121  {
122  Emit(DefaultInstructionSet.Instance.Dig(1).CreateStk(preds, 2, otype0, otype1, otype1, otype0));
123  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(1, otype0, rtype));
124  Emit(DefaultInstructionSet.Instance.Dig(1).CreateStk(preds, 2, otype1, rtype, rtype, otype1));
125  preds = new InstructionDependency[0];
126  }

128  {
129  Emit(DefaultInstructionSet.Instance.Convert().CreateStk(preds, 1, otype1, rtype));
130  }
131  otypes[0] = rtype;
132  otypes[1] = rtype;
133  }
134  }
135  else
136  {
137  otypes[0] = i.OperandTypes[0];
138  otypes[1] = i.OperandTypes[1];
139  }

143  private TypeDescriptor GetNativeResultType(XILSInstr i, TypeDescriptor[] otypes)
144  {
145  Contract.Requires<ArgumentNullException>(i != null);
146  Contract.Requires<ArgumentNullException>(otypes != null);
147  Contract.Requires<ArgumentException>(i.OperandTypes.Length == otypes.Length);
148

154  DesignContext.Instance.FixPoint.ArithSizingMode = EArithSizingMode.VHDLCompliant;
155  switch (i.Name)

157  case InstructionCodes.Add:
158  rexpr = exprs[0] + exprs[1];
159  break;
160 
161  case InstructionCodes.Sub:
162  rexpr = exprs[0] - exprs[1];
163  break;
164 
165  case InstructionCodes.Mul:
166  rexpr = exprs[0] * exprs[1];
167  break;
168 
169  case InstructionCodes.Div:
170  rexpr = exprs[0] / exprs[1];
171  break;
172 
173  case InstructionCodes.Neg:
174  rexpr = -exprs[0];
175  break;
176 
177  default:
178  throw new NotImplementedException();
179  }
180  var result = rexpr.ResultType;
181  DesignContext.Instance.FixPoint.ArithSizingMode = oldMode;

208  var outfmt = i.ResultTypes[0].GetFixFormat();

226  otype = SFix.MakeType(2, infmt.FracWidth);

229  }

231  var rtype = i.ResultTypes[0];
232  if (outfmt.IntWidth != 2)
233  {
234  // we gonna need exactly 2 integer bits for results
235  rtype = SFix.MakeType(2, outfmt.FracWidth);
236  }
237  Emit(DefaultInstructionSet.Instance.ScSinCos().CreateStk(
238  preds, 1, otype, rtype, rtype));

246  Emit(DefaultInstructionSet.Instance.Swap().CreateStk(
