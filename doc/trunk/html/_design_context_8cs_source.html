166  private static Action Resolve(Action cur, Action plus)
167  {
168  Contract.Requires(cur.GetInvocationList() != null);
169 
170  var list1 = plus.GetInvocationList();
171  var list2 = cur.GetInvocationList();
172  for (int i = 0; i < list1.Length; i++)
173  {
174  bool found = false;
175  for (int j = 0; j < list2.Length; j++)
176  {
177  if (list1[i].Target == list2[j].Target)
178  {
179  found = true;
180  break;
181  }
182  }
183  if (!found)
184  cur = (Action)Delegate.Combine(cur, list1[i]);
185  }
186 
187  return cur;
188  }
