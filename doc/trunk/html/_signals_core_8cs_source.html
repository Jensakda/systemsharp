716  public static SignalBase CreateInstance(object initialValue)
717  {
718  TypeDescriptor elemType = TypeDescriptor.GetTypeOf(initialValue);
719  SignalBase sinst;
720  if (elemType.CILType.Equals(typeof(StdLogic)))
721  {
722  sinst = new SLSignal();
723  sinst.InitialValueObject = initialValue;
724  }
725  else if (elemType.CILType.Equals(typeof(StdLogicVector)))
726  {
727  sinst = new SLVSignal(((StdLogicVector)initialValue).Size);
728  sinst.InitialValueObject = initialValue;
729  }
730  else if (elemType.CILType.IsArray && elemType.CILType.GetArrayRank() == 1)
731  {
732  Array array = (Array)initialValue;
733  Type subElemType = elemType.CILType.GetElementType();
734  int length = array.GetLength(0);
735  Array signalArray = Array.CreateInstance(typeof(Signal<>).MakeGenericType(subElemType), length);
736  for (int i = 0; i < length; i++)
737  {
738  signalArray.SetValue(CreateInstance(array.GetValue(i)), i);
739  }
740  sinst = (SignalBase)Activator.CreateInstance(
741  typeof(Signal1D<>).MakeGenericType(subElemType), signalArray);
742  }
743  else if (elemType.CILType.IsArray)
744  {
745  throw new NotSupportedException("Signals of multi-dimensional arrays not supported");
746  }
747  else
748  {
749  sinst = (SignalBase)Activator.CreateInstance(typeof(Signal<>).MakeGenericType(elemType.CILType));
750  sinst.InitialValueObject = initialValue;
751  }
752  return sinst;
