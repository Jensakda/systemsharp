149  public static double ToFloat(this StdLogicVector slv, FloatFormat fmt)
150  {
151  if (slv.Size != fmt.TotalWidth)
152  throw new ArgumentException("Vector does not match specified floating point format");
153  slv = slv.ProperValue;
154 
155  StdLogicVector mantissa = slv[fmt.FractionWidth - 1, 0];
156  StdLogicVector exponent = slv[fmt.FractionWidth + fmt.ExponentWidth - 1, fmt.FractionWidth];
157  StdLogic sign = slv[fmt.FractionWidth + fmt.ExponentWidth];
158 
159  int exp = (int)exponent.ULongValue - fmt.Bias;
160 
161  if (exponent.Equals(StdLogicVector._0s(fmt.ExponentWidth)))
162  {
163  // denormalized
164  long mant = mantissa.LongValue;
165  double result = (double)mant * Math.Pow(2.0, exp - 1);
166  return result;
167  }
168  else if (exponent.Equals(StdLogicVector._1s(fmt.ExponentWidth)))
169  {
170  // Infinity / NaN
171  if (mantissa.Equals(StdLogicVector._0s(fmt.FractionWidth)))
172  {
173  // infinity
174  if (sign == '1')
175  return double.NegativeInfinity;
176  else
177  return double.PositiveInfinity;
178  }
179  else
180  {
181  // NaN
182  return double.NaN;
183  }
184  }
185  else
186  {
187  // normalized
188  StdLogicVector number = StdLogicVector._1s(1).Concat(mantissa);
189  ulong mant = number.ULongValue;
190  double result = (double)mant * Math.Pow(2.0, exp - fmt.FractionWidth);
191  if (sign == '1')
192  result = -result;
193  return result;
194  }
195  }

215  public static StdLogicVector ToSLV(this double value, FloatFormat fmt)
216  {
217  StdLogicVector sign;
218  StdLogicVector exponent;
219  StdLogicVector mantissa;
220  if (double.IsInfinity(value))
221  {
222  sign = double.IsNegativeInfinity(value) ? (StdLogicVector)"1" : (StdLogicVector)"0";
223  exponent = StdLogicVector._1s(fmt.ExponentWidth);
224  mantissa = StdLogicVector._0s(fmt.FractionWidth);
225  }
226  else if (double.IsNaN(value))
227  {
228  sign = (StdLogicVector)"0";
229  exponent = StdLogicVector._1s(fmt.ExponentWidth);
230  mantissa = StdLogicVector._1s(fmt.FractionWidth);
231  }
232  else
233  {
234  sign = value < 0.0 ? (StdLogicVector)"1" : (StdLogicVector)"0";
235  double absvalue = Math.Abs(value);
236  int exp = 0;
237  while (absvalue >= 2.0)
238  {
239  absvalue *= 0.5;
240  exp++;
241  }
242  while (absvalue > 0.0 && absvalue < 1.0)
243  {
244  absvalue *= 2.0;
245  exp--;
246  }
247  if (absvalue == 0.0)
248  {
249  return StdLogicVector._0s(fmt.TotalWidth);
250  }
251  else if (exp <= -fmt.Bias)
252  {
253  // denomalized
254  exponent = StdLogicVector._0s(fmt.ExponentWidth);
255  absvalue *= (double)(1L << (fmt.FractionWidth+1));
256  long mant = (long)absvalue;
257  mantissa = StdLogicVector.FromLong(mant, fmt.FractionWidth);
258  }
259  else
260  {
261 
262  absvalue -= 1.0;
263  absvalue *= (double)(1L << fmt.FractionWidth);
264  long mant = (long)absvalue;
265  mantissa = StdLogicVector.FromLong(mant, fmt.FractionWidth);
266  exponent = StdLogicVector.FromLong(exp + fmt.Bias, fmt.ExponentWidth);
267  }
268  }
269  return sign.Concat(exponent.Concat(mantissa));
270  }
