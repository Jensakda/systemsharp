639  var lrClk = new LiteralReference(srClk);
640  var srEn = SignalRef.Create(_taSite._en, SignalRef.EReferencedProperty.Cur);

655  typeof(StdLogicVector),
656  TypeDescriptor.GetTypeOf(_taSite._port.InitialValue));
657  }
658  var srNext = SignalRef.Create(_taSite._port, SignalRef.EReferencedProperty.Next);
659 
660  If(cond);
661  {
662  Store(srNext, conv);
663  }
664  EndIf();
665  }
