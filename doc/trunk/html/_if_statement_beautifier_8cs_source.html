153  public override void AcceptIf(IfStatement stmt)
154  {
155  if (stmt.Conditions.Count == 1 &&
156  stmt.Branches.Count == 2 &&
157  stmt.Branches[0].IsEmpty())
158  {
159  /* This case models the following situation:
160  *
161  * if (someCondition)
162  * {
163  * // empty
164  * }
165  * else
166  * {
167  * someActions();
168  * }
169  *
170  * Thus, it is rewritten to the following statement:
171  *
172  * if (!someCondition)
173  * {
174  * someActions();
175  * }
176  * */
177 
178  If((!stmt.Conditions[0]).Simplify());
179  {
180  stmt.Branches[1].Accept(this);
181  }
182  EndIf();
183  }
184  else
185  {
186  Expression c0 = stmt.Conditions[0].Simplify();
187  Matching x = new Matching();
188  Matching not_x = !x;
189  if (c0.Match(not_x) == c0 &&
190  stmt.Conditions.Count == 1 &&
191  stmt.Branches.Count == 2)
192  {
193  /* This case models the following situation:
194  *
195  * if (!someCondition)
196  * {
197  * someThenAction();
198  * }
199  * else
200  * {
201  * someElseAction();
202  * }
203  *
204  * Thus, it is rewritten to the following statement:
205  *
206  * if (someCondition)
207  * {
208  * someElseAction();
209  * }
210  * else
211  * {
212  * someThenAction();
213  * }
214  * */
215  If(c0.Children[0]);
216  {
217  stmt.Branches[1].Accept(this);
218  }
219  Else();
220  {
221  stmt.Branches[0].Accept(this);
222  }
223  EndIf();
224  }
225  else
226  {
227  If(c0);
228  {
229  stmt.Branches[0].Accept(this);
230  }
231  for (int i = 1; i < stmt.Conditions.Count; i++)
232  {
233  ElseIf(stmt.Conditions[i].Simplify());
234  {
235  stmt.Branches[i].Accept(this);
236  }
237  }
238  if (stmt.Branches.Count > stmt.Conditions.Count)
239  {
240  Else();
241  {
242  stmt.Branches.Last().Accept(this);
243  }
244  }
245  EndIf();
246  }
247  }
248  CopyAttributesToLastStatement(stmt);
249  }
