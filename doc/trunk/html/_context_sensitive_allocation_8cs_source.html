210  public EAllocationDecision SelectBestMapping(XIL3Instr instr, long cstep, IEnumerable<IXILMapping> mappings, out IXILMapping bestMapping)
211  {
212  foreach (int rslot in instr.ResultSlots)
213  {
214  _interlinks[rslot] = new Interlink(rslot);
215  }
216 
217  bestMapping = null;
218  double bestCost = double.MaxValue;
219 
220  foreach (var mapping in mappings)
221  {
222  var sources = instr.OperandSlots.Select(os => _interlinks[os]).ToArray();
223  var sinks = instr.ResultSlots.Select(rs => _interlinks[rs]).ToArray();
224 
225  var verbs = mapping.Realize(sources, sinks);
226  long curStep = cstep;
227  double cost = 0.0;
228  foreach (var verb in verbs)
229  {
230  var pflow = verb.ToCombFlow();
231  foreach (var flow in pflow.Flows)
232  {
233  var sflow = flow as SignalFlow;
234  if (sflow == null)
235  continue;
236 
237  if (sflow.Source.Desc.HasAttribute<int>() &&
238  !sflow.Target.Desc.HasAttribute<int>())
239  {
240  int index = sflow.Source.Desc.QueryAttribute<int>();
241  var driver = _interlinks[index].Driver;
242  if (driver == null)
243  continue;
244 
245  var fanIn = _fanIn[sflow.Target.Desc];
246  foreach (int v in fanIn.Values)
247  {
248  cost += Math.Pow(2.0, -v);
249  }
250  int prevSlack;
251  if (fanIn.TryGetValue(driver, out prevSlack))
252  {
253  cost -= Math.Pow(2.0, -prevSlack);
254  }
255  int slack = (int)(curStep - _interlinks[index].DriveTime);
256  cost += Math.Pow(2.0, -slack);
257  }
258  }
259 
260  curStep++;
261  }

264

268  bestMapping = mapping;

270  }
