211  public void Synthesize()
212  {
213  var repSet = new Dictionary<Component, Component>(Component.BehaviorComparer);
214  foreach (var cdesc in _componentSet)
215  {
216  var cd = cdesc as ComponentDescriptor;
217  if (cd == null)
218  continue;
219  var component = cd.Instance;
220  if (repSet.ContainsKey(component))
221  component.Representant = repSet[component];
222  else
223  repSet[component] = component;
224  }
225 
226  foreach (Component component in repSet.Keys)
227  {
228  component.Descriptor.HasForeignImplementation = true;
229  component.OnSynthesisInternal(this);
230  }
231  ModelContext.CompleteAnalysis();
232  ModelContext.Descriptor.ResolveTypeDependencies();
233  CodeGen.Initialize(Project, ModelContext);
234  foreach (var component in repSet.Keys)
235  {
236  var cdd = component.Descriptor;
237  if (cdd != null && cdd.HasForeignImplementation)
238  continue;
239  CodeGen.GenerateComponent(Project, cdd);
240  }
241  foreach (PackageDescriptor pd in ModelContext.Descriptor.GetPackages())
242  {
243  if (pd.IsEmpty)
244  continue;
245  CodeGen.GeneratePackage(Project, pd);
246  }
247  }
