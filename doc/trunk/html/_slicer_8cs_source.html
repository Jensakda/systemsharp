149  public Slicer(int inputWidth, int hiOffset, int loOffset, bool signed)
150  {
151  Contract.Requires<ArgumentOutOfRangeException>(hiOffset - loOffset >= -1);
152 
153  InputWidth = inputWidth;
154  HiOffset = hiOffset;
155  LoOffset = loOffset;
156  IsSigned = signed;
157  TASite = new SlicerTransactionSite(this);
158 
159  int msb = signed ? inputWidth - 2 : inputWidth - 1;
160  int hiPadWidth = Math.Max(hiOffset, msb) - Math.Max(loOffset, msb + 1) + 1;
161  int loPadWidth = Math.Min(hiOffset, -1) - Math.Min(loOffset, 0) + 1;
162  _hiSlice = Math.Min(msb, hiOffset);
163  _loSlice = Math.Max(0, loOffset);
164 
165  if (_loSlice > _hiSlice)
166  {
167  // degenerate case: actually no portion of input word is used
168  _hiSlice = -1;
169  _loSlice = 0;
170  }
171 
172  _hiPad0 = StdLogicVector._0s(hiPadWidth);
173  _hiPad1 = StdLogicVector._1s(hiPadWidth);
174  _loPad = StdLogicVector._0s(loPadWidth);
175 
176  Debug.Assert(hiPadWidth + loPadWidth + _hiSlice - _loSlice == hiOffset - loOffset);
177  }

324  private bool GetSliceParams(XILInstr instr, TypeDescriptor[] operandTypes, TypeDescriptor[] resultTypes,
325  out bool isSigned, out int inputWidth, out int hiOffset, out int loOffset)
326  {
327  isSigned = false;
328  inputWidth = 0;
329  hiOffset = 0;
330  loOffset = 0;
331 
332  if (instr.Name != InstructionCodes.Convert &&
333  instr.Name != InstructionCodes.SliceFixI &&
334  instr.Name != InstructionCodes.Rempow2)
335  return false;
336 
337  bool o0fix = IsSigned(operandTypes[0]) || IsUnsigned(operandTypes[0]);
338  bool r0fix = IsSigned(resultTypes[0]) || IsUnsigned(resultTypes[0]);
339  bool u2s = o0fix && r0fix;
340  bool typesEq = operandTypes[0].CILType.Equals(resultTypes[0].CILType);
341  if (!u2s && !typesEq)
342  return false;
343  if (!IsSliceable(operandTypes[0]) || !IsSliceable(resultTypes[0]))
344  return false;
345  isSigned = IsSigned(operandTypes[0]) && IsSigned(resultTypes[0]);
346 
347  inputWidth = TypeLowering.Instance.GetWireWidth(operandTypes[0]);
348 
349  switch (instr.Name)
350  {
351  case InstructionCodes.Convert:
352  {
353  TypeDescriptor inWType = operandTypes[0];
354  TypeDescriptor outWType = resultTypes[0];
355  hiOffset = outWType.Constraints[0].FirstBound - inWType.Constraints[0].SecondBound;
356  loOffset = outWType.Constraints[0].SecondBound - inWType.Constraints[0].SecondBound;
357  }
358  break;
359 
360  case InstructionCodes.SliceFixI:
361  {
362  Range range = (Range)instr.Operand;
363  hiOffset = range.FirstBound;
364  loOffset = range.SecondBound;
365  }
366  break;
367 
368  case InstructionCodes.Rempow2:
369  {
370  if (!IsSigned(operandTypes[0]) ||
371  !IsSigned(resultTypes[0]))
372  return false;
373 
374  var infmt = operandTypes[0].GetFixFormat();
375  int p2 = (int)instr.Operand;
376  if (infmt.IntWidth > p2 + 1)
377  return false;
378 
379  goto case InstructionCodes.Convert;
380  }
381 
382  default:
383  throw new NotImplementedException();
384  }
385 
386  return true;
387  }
