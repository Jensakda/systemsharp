564  public int TransformFunction(FunctionCall expr)
565  {
566  FunctionSpec fspec = expr.Callee as FunctionSpec;
567  if (fspec == null)
568  throw new NotSupportedException();
569 
570  IntrinsicFunction ifun = fspec.IntrinsicRep;
571  if (ifun == null)
572  throw new NotSupportedException();
573 
574  switch (ifun.Action)
575  {
576  case IntrinsicFunction.EAction.Convert:
577  {
578  // only first child should be evaluated, following children are conversion arguments.
579  // These will be implicitly defined by the result type of the instruction
580  expr.Children[0].Accept(this);
581  ExtractCILIndex(expr);
582  var cparams = (CastParams)ifun.Parameter;
583  Emit(ISet.Convert(cparams.Reinterpret), expr, 1, expr.ResultType);
584  }
585  return 0;
586 
587  case IntrinsicFunction.EAction.Sign:
588  expr.Children[0].Accept(this);
589  ExtractCILIndex(expr);
590  Emit(ISet.Sign(), expr, 1, expr.ResultType);
591  return 0;
592 
593  case IntrinsicFunction.EAction.Resize:
594  expr.Children[0].Accept(this);
595  Emit(ISet.Convert(), expr, 1, expr.ResultType);
596  ExtractCILIndex(expr);
597  return 0;
598 
599  case IntrinsicFunction.EAction.Slice:
600  if (expr.Children.Length == 3)
601  {
602  expr.Children[0].Accept(this);
603  expr.Children[1].Accept(this);
604  expr.Children[2].Accept(this);
605  Emit(ISet.Slice(), expr, 3, expr.ResultType);
606  }
607  else
608  {
609  expr.Children[0].Accept(this);
610  var range = (Range)ifun.Parameter;
611  Emit(ISet.SliceFixI(range), expr, 1, expr.ResultType);
612  }
613  return 0;
614 
615  case IntrinsicFunction.EAction.Abs:
616  expr.Children[0].Accept(this);
617  Emit(ISet.Abs(), expr, 1, expr.ResultType);
618  return 0;
619 
620  case IntrinsicFunction.EAction.Sqrt:
621  expr.Children[0].Accept(this);
622  Emit(ISet.Sqrt(), expr, 1, expr.ResultType);
623  return 0;
624 
625  case IntrinsicFunction.EAction.GetArrayElement:
626  {
627  var aex = expr.Children[0];
628  var alr = (LiteralReference)aex;
629  var alit = alr.ReferencedObject;
630  var arr = (Array)aex.ResultType.GetSampleInstance();
631  var far = new FixedArrayRef(alit, arr);
632  expr.Children[1].Accept(this);
633  Emit(ISet.LdelemFixA(far), expr, 1, expr.ResultType);
634  }
635  return 0;
636 
637  case IntrinsicFunction.EAction.XILOpCode:
638  {
639  foreach (var child in expr.Children)
640  child.Accept(this);
641  var opcode = (XILInstr)ifun.Parameter;
642  Emit(opcode, expr, expr.Children.Length, expr.ResultType.Unpick());
643  }
644  return 0;
645 
646  case IntrinsicFunction.EAction.TupleSelect:
647  {
648  int item = (int)ifun.Parameter;
649  var lr = expr.Children[0] as LiteralReference;
650  if (lr == null)
651  throw new NotSupportedException("TupleSelect must refer to a literal");
652  var tuple = lr.ReferencedObject as Variable;
653  if (tuple == null)
654  throw new NotSupportedException("TupleSelect must refer to a local variable");
655  var itemVars = Unpick(tuple);
656  var itemLr = new LiteralReference(itemVars[item]);
657  TransformLiteralReference(itemLr);
658  }
659  return 0;
660 
661  default:
662  throw new NotImplementedException();
663  }

698  public void VisitVariable(Variable variable)
699  {
700  if (variable.Type.HasIntrinsicTypeOverride &&
701  variable.Type.IntrinsicTypeOverride == EIntrinsicTypes.Tuple)
702  {
703  var itemVars = Unpick(variable);
704  foreach (var itemVar in itemVars.Reverse())
705  {
706  VisitVariable(itemVar);
707  }
708  }
709  else
710  {
711  InstructionDependency[] preds;
712  switch (_litMode)
713  {
714  case ELiteralAcceptMode.Read:
715  preds = _storesInCurBB.Get(variable).Select(
716  _ => new OrderDependency(_, OrderDependency.EKind.BeginAfter)).ToArray();
717  _readsInCurBB.Add(variable, NextInstructionIndex);
718  Emit(ISet.LoadVar(variable), _curExpr, preds, 0, variable.Type);
719  break;
720 
721  case ELiteralAcceptMode.Write:
722  {
723  preds = _storesInCurBB.Get(variable)
724  .Union(_readsInCurBB.Get(variable))
725  .Select(_ => new OrderDependency(_, OrderDependency.EKind.BeginAfter))
726  .ToArray();
727  if (preds.Any())
728  {
729  int idx = NextInstructionIndex;
730  Emit(DefaultInstructionSet.Instance.Nop(0), _curExpr, preds, 0);
731  preds = new InstructionDependency[] { new OrderDependency(idx, OrderDependency.EKind.BeginAfter) };
732  }
733  _storesInCurBB.Add(variable, NextInstructionIndex);
734  Debug.Assert(variable.Type.Equals(_typeStack.Peek()));
735  Emit(ISet.StoreVar(variable), _curExpr, preds, 1);
736  }
737  break;
738  }

858  public void VisitArrayRef(ArrayRef arrayRef)
859  {
860  LiteralReference lr = arrayRef.ArrayExpr as LiteralReference;
861  object arrayObj;
862  if (lr == null || !lr.ReferencedObject.IsConst(out arrayObj))
863  throw new NotSupportedException("Array references must be fixed (referenced array must be known in advance)!");

865  if (array == null)
866  throw new InvalidOperationException("Not an array");
867 
868  ELiteralAcceptMode saveMode = _litMode;
869  _litMode = ELiteralAcceptMode.Read;
870  FixedArrayRef far = arrayRef.AsFixed();
871  if (!far.IndicesConst)
872  {
873  foreach (Expression index in arrayRef.Indices)
874  {
875  index.Accept(this);
876  }
877  }
878  _litMode = saveMode;
879 
880  int[] preds;
881  switch (_litMode)
882  {
883  case ELiteralAcceptMode.Read:
884  if (far.IndicesConst)
885  {
886  ArrayMod amod1 = new ArrayMod(array, far.Indices);
887  ArrayMod amod2 = new ArrayMod(array, ArrayMod.EMode.UnknownIndex);
888  ArrayMod amod3 = new ArrayMod(array, ArrayMod.EMode.AnyIndex);
889  preds = _storesInCurBB.Get(amod1)
890  .Union(_storesInCurBB.Get(amod2))
891  .Union(_storesInCurBB.Get(amod3))
892  .Union(_readsInCurBB.Get(amod3))
893  .Distinct()
894  .ToArray();
895  _readsInCurBB.Add(amod1, NextInstructionIndex);
896  _readsInCurBB.Add(amod3, NextInstructionIndex);
897  Emit(ISet.LdelemFixAFixI(far), preds, 0, arrayRef.Type);
898  }
899  else
900  {
901  ArrayMod amod1 = new ArrayMod(array, ArrayMod.EMode.UnknownIndex);
902  ArrayMod amod2 = new ArrayMod(array, ArrayMod.EMode.AnyIndex);
903  preds = _storesInCurBB.Get(amod1)
904  .Union(_storesInCurBB.Get(amod2))
905  .Union(_readsInCurBB.Get(amod2))
906  .Distinct()
907  .ToArray();
908  _readsInCurBB.Add(amod1, NextInstructionIndex);
909  Emit(ISet.LdelemFixA(far), preds, arrayRef.Indices.Length, arrayRef.Type);
910  }
911  break;
912 
913  case ELiteralAcceptMode.Write:
914  if (far.IndicesConst)
915  {
916  ArrayMod amod1 = new ArrayMod(array, far.Indices);
917  ArrayMod amod2 = new ArrayMod(array, ArrayMod.EMode.UnknownIndex);
918  ArrayMod amod3 = new ArrayMod(array, ArrayMod.EMode.AnyIndex);
919  preds = _storesInCurBB.Get(amod1)
920  .Union(_storesInCurBB.Get(amod3))
921  .Union(_storesInCurBB.Get(amod2))
922  .Union(_readsInCurBB.Get(amod1))
923  .Union(_readsInCurBB.Get(amod2))
924  .Union(_readsInCurBB.Get(amod3))
925  .Distinct()
926  .ToArray();

928  _storesInCurBB.Add(amod3, NextInstructionIndex);

930  }
931  else

934  ArrayMod amod2 = new ArrayMod(array, ArrayMod.EMode.AnyIndex);
935  preds = _storesInCurBB.Get(amod1)
936  .Union(_storesInCurBB.Get(amod2))
937  .Union(_readsInCurBB.Get(amod1))
938  .Union(_readsInCurBB.Get(amod2))
939  .Distinct()

942  Emit(ISet.StelemFixA(far), preds, 1 + arrayRef.Indices.Length);
943  }
944  break;
945  }
