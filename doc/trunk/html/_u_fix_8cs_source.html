171  private static FixFormat Equalize(UFix a, UFix b, out Unsigned an, out Unsigned bn)
172  {
173  an = a._value;
174  bn = b._value;
175  FixFormat dfmt;
176  switch (DesignContext.Instance.FixPoint.ArithSizingMode)
177  {
178  case EArithSizingMode.Safe:
179  dfmt = new FixFormat(false,
180  Math.Max(a.Format.IntWidth, b.Format.IntWidth) + 1,
181  Math.Max(a.Format.FracWidth, b.Format.FracWidth));
182  an = an.Resize(dfmt.TotalWidth - 1);
183  bn = bn.Resize(dfmt.TotalWidth - 1);
184  if (a.Format.FracWidth > b.Format.FracWidth)
185  bn = bn << (int)(a.Format.FracWidth - b.Format.FracWidth);
186  else if (b.Format.FracWidth > a.Format.FracWidth)
187  an = an << (int)(b.Format.FracWidth - a.Format.FracWidth);
188  return dfmt;
189 
190  case EArithSizingMode.VHDLCompliant:
191  dfmt = new FixFormat(false,
192  Math.Max(a.Format.IntWidth, b.Format.IntWidth) + 1,
193  Math.Max(a.Format.FracWidth, b.Format.FracWidth));
194  an = an.Resize(dfmt.TotalWidth);
195  bn = bn.Resize(dfmt.TotalWidth);
196  if (a.Format.FracWidth > b.Format.FracWidth)
197  bn = bn << (int)(a.Format.FracWidth - b.Format.FracWidth);
198  else if (b.Format.FracWidth > a.Format.FracWidth)
199  an = an << (int)(b.Format.FracWidth - a.Format.FracWidth);
200  return dfmt;
201 
202  case EArithSizingMode.InSizeIsOutSize:
203  dfmt = a.Format;
204  bn = bn.Resize(dfmt.TotalWidth);
205  if (a.Format.FracWidth > b.Format.FracWidth)
206  bn = bn << (int)(a.Format.FracWidth - b.Format.FracWidth);
207  else if (b.Format.FracWidth > a.Format.FracWidth)
208  an = an << (int)(b.Format.FracWidth - a.Format.FracWidth);
209  return dfmt;
210 
211  default:
212  throw new NotImplementedException();
213  }
