278  public static Attribute[] GetInjectedAttributes(this MethodBase method, Type type)
279  {
280  var r1 = _methodAttrs.GetOrCreateValue(method);
281  var r2 = method.DeclaringType.GetBaseTypeChain().SelectMany(
282  t => _methodByNameAttrs.GetOrCreateValue(t).Get(method.Name));
283  var mi = method as MethodInfo;
284  var r3 = method.IsGenericMethod ?
285  _methodAttrs.GetOrCreateValue(mi.GetGenericMethodDefinition()) :
286  Enumerable.Empty<Attribute>();
287  var r4 = Enumerable.Empty<Attribute>();
288  if (method.DeclaringType.IsGenericType &&
289  !method.DeclaringType.IsGenericTypeDefinition)
290  {
291  var mtype = method.DeclaringType.GetGenericTypeDefinition();
292  BindingFlags flags = BindingFlags.Default;
293  if (method.CallingConvention.HasFlag(CallingConventions.HasThis))
294  flags |= BindingFlags.Instance;
295  if (method.IsPublic)
296  flags |= BindingFlags.Public;
297  else
298  flags |= BindingFlags.NonPublic;
299 
300  //FIXME: Disambiguate by parameter types
301  MethodBase gmethod;
302  if (method.IsConstructor)
303  {
304  flags |= BindingFlags.CreateInstance;
305  gmethod = mtype.GetConstructors(flags)
306  .FirstOrDefault();
307  }
308  else
309  {
310  gmethod = mtype.GetMethods(flags)
311  .Where(m => m.Name == method.Name)
312  .FirstOrDefault();
313  }
314  if (gmethod == null)
315  r4 = Enumerable.Empty<Attribute>();
316  else
317  r4 = GetInjectedAttributes(gmethod, type);
318  }
319  return SelectAttributes(r1.Concat(r2).Concat(r3).Concat(r4), type).ToArray();
320  }
