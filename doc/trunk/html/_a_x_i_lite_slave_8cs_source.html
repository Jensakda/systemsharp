304  private void WriteProcess()
305  {
306  if (Bus2IP_Clk.RisingEdge())
307  {
308  if (Bus2IP_Resetn.Cur == '0')
309  {
310  for (int i = 0; i < NumWriteSignals; i++)
311  {
312  ProgramFlow.Unroll();
313  GetWriteSignal(i).NextObject = GetWriteSignal(i).InitialValueObject;
314  }
315  }
316  else
317  {
318  for (int reg = 0; reg < NumRegs; reg++)
319  {
320  ProgramFlow.Unroll();
321 
322  if (Bus2IP_WrCE.Cur[NumRegs - reg - 1] == '1')
323  {
324  for (int byteIdx = 0; byteIdx < SLVDWidth / 8; byteIdx++)
325  {
326  ProgramFlow.Unroll();
327 
328  if (Bus2IP_BE.Cur[byteIdx] == '1')
329  {
330  for (int bitIdx = 0; bitIdx < 8; bitIdx++)
331  {
332  ProgramFlow.Unroll();
333  if (IsWriteBitPresent(reg, byteIdx, bitIdx))
334  GetWriteBit(reg, byteIdx, bitIdx).Next = Bus2IP_Data.Cur[8 * byteIdx + bitIdx];
335  }
336  }
337  }
338  }
339  }
340  }
