410  private bool Merge(VariabilityInfo a, VariabilityInfo b, out VariabilityInfo m)
411  {
412  m = VariabilityInfo.MergeDefs(a, b);
413  if (m.Variability == EVariability.Constant && m.Definitions.Count() > 1)
414  {

445  protected override bool Merge(
446  AbstractStackState<VariabilityInfo> a,
447  AbstractStackState<VariabilityInfo> b,
448  out AbstractStackState<VariabilityInfo> m)
449  {
450  List<VariabilityInfo> stack = new List<VariabilityInfo>();
451  bool changed = Merge(i => a[i], i => b[i], stack, a.Depth);
452  List<VariabilityInfo> locals = new List<VariabilityInfo>();
453  if (Merge(i => a.GetLocal(i), i => b.GetLocal(i), locals, a.NumLocals))
454  changed = true;
455  List<VariabilityInfo> args = new List<VariabilityInfo>();
456  if (Merge(i => a.GetArgument(i), i => b.GetArgument(i), args, a.NumArguments))
457  changed = true;
458  if (a.HasUniqueSuccessor() != b.HasUniqueSuccessor())
459  changed = true;
460  m = new IndependentStackState(stack, locals, args,
461  a.HasUniqueSuccessor() && b.HasUniqueSuccessor());

541  {
542  MethodBase callee = (MethodBase)ili.Operand;
543  bool hasThis = callee.CallingConvention.HasFlag(CallingConventions.HasThis);
544  EVariability callVar = EVariability.Constant;
545  ParameterInfo[] args = callee.GetParameters();
546  MethodFacts myFacts = FactUniverse.Instance.GetFacts(Method);
547  MethodFacts calleeFacts = FactUniverse.Instance.GetFacts(callee);
548  AbstractStackState<VariabilityInfo> next = pre;

555  {
556  callVar = VariabilityOperations.Stronger(callVar, next[0].Variability);

566  next = UpdateStackState(ili, callVarI, next);
