297  public int CreateTable(Type ty)
298  {
299  if (_tables == null) {
300  _tables = new Dictionary<string, TableMapping> ();
301  }
302  TableMapping map;
303  if (!_tables.TryGetValue (ty.FullName, out map)) {
304  map = GetMapping (ty);
305  _tables.Add (ty.FullName, map);
306  }
307  var query = "create table if not exists \"" + map.TableName + "\"(\n";
308 
309  var decls = map.Columns.Select (p => Orm.SqlDecl (p, StoreDateTimeAsTicks));
310  var decl = string.Join (",\n", decls.ToArray ());
311  query += decl;
312  query += ")";
313 
314  var count = Execute (query);
315 
316  if (count == 0) { //Possible bug: This always seems to return 0?
317  // Table already exists, migrate it
318  MigrateTable (map);
319  }

322  foreach (var c in map.Columns) {
323  foreach (var i in c.Indices) {
324  var iname = i.Name ?? map.TableName + "_" + c.Name;
325  IndexInfo iinfo;
326  if (!indexes.TryGetValue (iname, out iinfo)) {
327  iinfo = new IndexInfo {
328  IndexName = iname,
329  TableName = map.TableName,
330  Unique = i.Unique,
331  Columns = new List<IndexedColumn> ()
332  };
333  indexes.Add (iname, iinfo);
334  }
335 
336  if (i.Unique != iinfo.Unique)
337  throw new Exception ("All the columns in an index must have the same value for their Unique property");
338 
339  iinfo.Columns.Add (new IndexedColumn {
340  Order = i.Order,
341  ColumnName = c.Name
342  });
343  }
344  }
345 
346  foreach (var indexName in indexes.Keys) {
347  var index = indexes[indexName];
348  const string sqlFormat = "create {3} index if not exists \"{0}\" on \"{1}\"(\"{2}\")";
349  var columns = String.Join("\",\"", index.Columns.OrderBy(i => i.Order).Select(i => i.ColumnName).ToArray());
350  var sql = String.Format (sqlFormat, indexName, index.TableName, columns, index.Unique ? "unique" : "");

717  public void BeginTransaction ()
718  {
719  // The BEGIN command only works if the transaction stack is empty,
720  // or in other words if there are no pending transactions.
721  // If the transaction stack is not empty when the BEGIN command is invoked,
722  // then the command fails with an error.
723  // Rather than crash with an error, we will just ignore calls to BeginTransaction
724  // that would result in an error.
725  if (Interlocked.CompareExchange (ref _trasactionDepth, 1, 0) == 0) {
726  try {
727  Execute ("begin transaction");
728  } catch (Exception ex) {
729  var sqlExp = ex as SQLiteException;
730  if (sqlExp != null) {
731  // It is recommended that applications respond to the errors listed below
732  // by explicitly issuing a ROLLBACK command.
733  // TODO: This rollback failsafe should be localized to all throw sites.
734  switch (sqlExp.Result) {
735  case SQLite3.Result.IOError:
736  case SQLite3.Result.Full:
737  case SQLite3.Result.Busy:
738  case SQLite3.Result.NoMem:
739  case SQLite3.Result.Interrupt:
740  RollbackTo (null, true);
741  break;
742  }
743  } else {
744  // Call decrement and not VolatileWrite in case we've already
745  // created a transaction point in SaveTransactionPoint since the catch.
746  Interlocked.Decrement (ref _trasactionDepth);
747  }
748 
749  throw;
750  }
751  } else {
752  // Calling BeginTransaction on an already open transaction is invalid
753  throw new System.InvalidOperationException ("Cannot begin a transaction while already in a transaction.");
754  }
755  }
