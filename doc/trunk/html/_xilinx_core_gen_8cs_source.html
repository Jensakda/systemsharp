392  public ProcessPool.ToolBatch Execute(XilinxProject proj, string xcoPath, string projPath)
393  {
394  if (proj.SkipIPCoreSynthesis)
395  return null;
396 
397  //see: http://www.xilinx.com/itp/xilinx6/books/docs/cgn/cgn.pdf
398  projPath = Path.GetFullPath(projPath);
399  xcoPath = Path.GetFullPath(xcoPath);
400  string dir = Path.GetDirectoryName(projPath);
401  bool madeSubst = false;
402  char drivel = default(char);
403  string drive = null;
404  try
405  {
406  var batch = ProcessPool.Instance.CreateBatch();
407  if (projPath.Length >= 160)
408  {
409  drivel = SubstManager.Instance.AllocateDrive();
410  drive = drivel + ":";
411  batch.Add("subst ", drive + " " + dir);
412  dir = drive + "\\";
413  projPath = dir + Path.GetFileName(projPath);
414  madeSubst = true;
415  }
416  string arguments = "-b \"" + xcoPath + "\" -p \"" + projPath + "\"";
417  batch.Add(proj.ISEBinPath, dir, "coregen", arguments);
418  if (madeSubst)
419  {
420  batch.Add("subst", drive + " /D");
421  }
422  batch.Start();
423  return batch;
424  }
425  finally
426  {
427  if (drivel != default(char))
428  SubstManager.Instance.ReleaseDrive(drivel);
429  }
430  }
