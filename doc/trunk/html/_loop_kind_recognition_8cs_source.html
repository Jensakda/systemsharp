519 
520  loop.Initializer = initializer;
521  loop.Body = newBody;
522  loop.Step = step;
523 
524  if (Level == EForLoopLevel.Strict ||

528

533  if (counterVar == null)

537

539  if (!counterType.IsEnumerable())
540  return;
541 
542  if (loop.Body.Modifies(counterVar))
543  return;
544 
545  loop.CounterStart = initStore.Value;
546

548  return;
549

552

569  else
