228  {

348  {
349  var fu = taSite.Host;
350  Shifter shifter = fu as Shifter;

352  yield break;
353 
354  if (instr.Name != InstructionCodes.LShift &&
355  instr.Name != InstructionCodes.RShift)
356  yield break;
357 
358  int fmtData = GetFixSize(operandTypes[0]);
359  int fmtShift = GetFixSize(operandTypes[1]);
360  int fmtResult = GetFixSize(resultTypes[0]);
361  if (fmtData < 0 || fmtShift < 0 || fmtResult < 0)
362  yield break;
363

365  yield break;
366 
367  if (shifter.DataWidth != fmtData ||
368  shifter.ShiftWidth != fmtShift)
369  yield break;

373  case InstructionCodes.LShift:
374  yield return new LShiftXILMapping(shifter);
375  break;
