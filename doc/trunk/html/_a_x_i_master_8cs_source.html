451  private void WriteProcess()
452  {
453  if (Bus2IP_Clk.RisingEdge())
454  {
455  if (Bus2IP_Resetn.Cur == '0')
456  {
457  for (int i = 0; i < NumWriteSignals; i++)
458  {
459  ProgramFlow.Unroll();
460  GetWriteSignal(i).NextObject = GetWriteSignal(i).InitialValueObject;
461  }
462  }
463  else
464  {
465  for (int reg = 0; reg < NumRegs; reg++)
466  {
467  ProgramFlow.Unroll();
468 
469  if (Bus2IP_WrCE.Cur[NumRegs - reg - 1] == '1')
470  {
471  for (int byteIdx = 0; byteIdx < SLVDWidth / 8; byteIdx++)
472  {
473  ProgramFlow.Unroll();
474

476  {

478  {
