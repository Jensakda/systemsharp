61  public override bool Rewrite(CodeDescriptor decompilee, MethodBase callee, StackElement[] args, IDecompiler stack, IFunctionBuilder builder)
62  {
63  var stateLookup = stack.QueryAttribute<IStateLookup>();
64  if (stateLookup == null)
65  {
66  var pd = decompilee as ProcessDescriptor;
67  if (pd == null)
68  {
69  var md = decompilee as MethodDescriptor;
70  if (md == null)
71  throw new InvalidOperationException("Unsupported code descriptor: " + decompilee);
72  pd = md.CallingProcess;

74  var pred = pd.Instance.Predicate;

76  pred.Method, new StackElement(LiteralReference.CreateConstant(pred.Target), pred.Target, EVariability.Constant));
