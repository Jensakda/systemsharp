79  if (flags.HasFlag(EXilinxIPCores.FixedPointDiv))

99  public int CalcALUPipelineDepth(ALU.EFunction op, ALU.EArithMode mode, int osize0, int osize1, int rsize)
100  {
101  switch (op)
102  {
103  case ALU.EFunction.Add:
104  case ALU.EFunction.Compare:
105  case ALU.EFunction.Sub:
106  {
107  float max = (float)osize0 * 0.0625f;
108  int depth = (int)Math.Round(_ratio * max);
109  return depth;
110  }
111 
112  case ALU.EFunction.And:
113  case ALU.EFunction.Not:
114  case ALU.EFunction.Or:
115  return 0;
116 
117  case ALU.EFunction.Mul:
118  {
119  float min = 0.0f;
120  // These coefficients were found experimentally by
121  // regression of Xilinx datasheet
122  float qterm = 0.0049f * (float)osize0 * (float)osize1;
123  float linterm = 0.0857f * (float)(osize0 + osize1);
124  float max = 0.5f * qterm + 0.4f * linterm + 1.0f;
125  int depth = (int)Math.Round(min + _ratio * (max - min));
126  return depth;
127  }
128 
129  case ALU.EFunction.Neg:
130  {
131  float max = osize0 / 16.0f;
132  int depth = (int)Math.Ceiling(_ratio * max);
133  return depth;
134  }
135 
136  default:
137  throw new NotImplementedException();
138  }
139  }
