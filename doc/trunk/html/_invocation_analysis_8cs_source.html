207  int numLocals = method.GetMethodBody().LocalVariables.Count;

209  ParameterInfo[] arguments = method.GetParameters();
210  IEnumerable<ElementSources> args =
211  arguments.Select(ai => Enumerable.Repeat(ElementSource.ForArgument(ai), 1));
212  bool hasThis = method.CallingConvention.HasFlag(CallingConventions.HasThis);
213  if (hasThis)
214  args = Enumerable.Repeat(Enumerable.Repeat((ElementSource)new ThisSource(method), 1), 1).Union(args);
215  _args = args.ToList();
216  }

562  private static bool Merge(Func<int, ElementSources> a,
563  Func<int, ElementSources> b,
564  int count,
565  out List<ElementSources> m)
566  {
567  bool changed = false;
568  m = new List<ElementSources>();
569  for (int i = 0; i < count; i++)
570  {
571  ElementSources ai = a(i);
572  ElementSources elems = ai.Union(b(i)).Distinct();
573  if (elems.Count() != ai.Count())
574  changed = true;
575  m.Add(elems.ToArray());
576  }
577  return changed;
578  }

580  protected override bool Merge(StackState a, StackState b, out StackState m)
581  {
582  int depth = a.Depth;
583 
584  List<ElementSources> stack = new List<ElementSources>();
585  List<ElementSources> locals = new List<ElementSources>();
586  List<ElementSources> args = new List<ElementSources>();
587  bool changed = Merge(i => a[i], i => b[i], depth, out stack);
588  if (Merge(i => a.GetLocal(i), i => b.GetLocal(i), a.NumLocals, out locals))
589  changed = true;
590  if (Merge(i => a.GetArgument(i), i => b.GetArgument(i), a.NumArguments, out args))
591  changed = true;
592  m = new IndependentStackState(stack, locals, args);
593  Debug.Assert(m.Depth == a.Depth);
594  Debug.Assert(m.NumLocals == a.NumLocals);
595  Debug.Assert(m.NumArguments == a.NumArguments);
596 
597  return changed;
598  }

693  private StackState HandleCall(ILInstruction ili, StackState pre, bool isCalli)
694  {
695  MethodBase callee = (MethodBase)ili.Operand;
696  PublishType(callee.DeclaringType);
697  PublishMethodCall(callee, ili.Index);
698  ParameterInfo[] args = callee.GetParameters();
699  int firstArg = callee.CallingConvention.HasFlag(CallingConventions.HasThis) ? -1 : 0;
700  int numParams = args.Length - firstArg;
701  int offs = isCalli ? -1 : 0;
702  int numElems = numParams - offs;
703  StackState next = pre;
704  for (int i = 0; i < numElems; i++)
705  {
706  ElementSources top = next[0];
707  int j = i + offs;
708  if (j >= 0)
709  {
710  int argIdx = args.Length - j - 1;
711  ElementSource rsource;
712  if (argIdx >= 0)
713  {
714  ParameterInfo arg = args[argIdx];
715  rsource = ElementSource.ForArgument(arg);
716  if (arg.ParameterType.IsByRef)
717  {
718  foreach (ElementSource source in top)
719  {
720  Debug.Assert(source is PointerSource);
721  PointerSource psource = (PointerSource)source;
722  ElementSource csource = new ArgumentReturnSource(arg);
723  IMutationSource msrc = psource as IMutationSource;
724  if (msrc != null)
725  {
726  ElementSources csources = Enumerable.Repeat(csource, 1);
727  foreach (ElementMutation mut in msrc.CreateMutations(ili.Index, arg.IsOut, csources))
728  {
729  if (mut is LocalMutation)
730  PublishLocalMutation((LocalMutation)mut);
731  if (mut is IModifiesStackState)
732  {
733  IModifiesStackState mss = (IModifiesStackState)mut;
734  next = mss.ModifyStackState(next, csources, (a, b) => a.Union(b).Distinct());
735  }
736  }
737  }
738  }
739  }
740  }
741  else
742  {
743  // "this"
744  rsource = new ThisSource(callee);
745  }
746  ElementSources lsources = top.SelectMany(s =>
747  {
748  if (s is AddressOfArgumentSource)
749  return pre.GetArgument(((AddressOfArgumentSource)s).Argument.Position);
750  else if (s is AddressOfLocalVariableSource)
751  return pre.GetLocal(((AddressOfLocalVariableSource)s).LocalIndex);
752  else
753  return Enumerable.Repeat(s, 1);
754  });
755  PublishEquivalence(lsources, rsource);
756  }
757  next = next.Pop();
758  }
759  Type returnType;
760  if (callee.IsFunction(out returnType))
761  {
762  next = next.Push(new MethodReturnSource((MethodInfo)callee));
763  }
764  return next;
765  }
