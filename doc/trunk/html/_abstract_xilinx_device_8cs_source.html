94  private void CreateUCF(XilinxProject proj)
95  {
96  string path = proj.AddFile(proj.ProjectName + ".ucf");
97  var sw = new StreamWriter(path);
98  foreach (var pin in PinList)
99  {
100  if (pin.AssociatedSignal == null)
101  continue;
102 
103  string indexExpr = "";
104  if (pin.AssociatedIndex.Length > 0)
105  indexExpr = pin
106  .AssociatedIndex
107  .Select(i => "[" + i + "]")
108  .Aggregate((x, y) => x + y);
109 
110  var sd = (ISignalOrPortDescriptor)((IDescriptive)pin.AssociatedSignal).Descriptor;
111  var pd = sd.AsSignalRef(SignalRef.EReferencedProperty.Instance)
112  .RelateToComponent(TopLevelComponent.Descriptor)
113  .Desc;
114 
115  sw.WriteLine("NET \"{0}{1}\" LOC = {2};",
116  pd.Name, indexExpr, pin.Name);
117  }
118  foreach (IPortDescriptor pd in TopLevelComponent.Descriptor.GetPorts())
119  {
120  var sd = pd.BoundSignal;
121  if (sd == null)
122  continue;
123 
124  var csa = sd.QueryAttribute<ClockSpecAttribute>();
125  if (csa == null)
126  continue;
127 
128  sw.WriteLine("TIMESPEC TS_{0} = PERIOD \"{1}\" {2} {3};",
129  pd.Name, pd.Name, csa.Period.Value, csa.Period.Unit);
130  sw.WriteLine("NET \"{0}\" TNM_NET = \"{1}\";",
131  pd.Name, pd.Name);
132  }
133  sw.Close();
134  }

146  IProject twinProject = null, EFlowStep step = EFlowStep.HDLGenAndIPCores)
147  {

149  XilinxProject project = new XilinxProject(destPath, designName)
150  {
151  TwinProject = twinProject
152  };
153 
154  project.ISEVersion = info.VersionTag;
155  if (info.Path == null)
156  project.SkipIPCoreSynthesis = true;
157  else
158  project.ISEBinPath = info.Path;
159 
160  project.PutProperty(EXilinxProjectProperties.DeviceFamily, Device.GetFamily());
161  project.PutProperty(EXilinxProjectProperties.Device, Device);
162  project.PutProperty(EXilinxProjectProperties.Package, Package);
163  project.PutProperty(EXilinxProjectProperties.SpeedGrade, SpeedGrade);
164  project.SetVHDLProfile();
165  if (!step.HasFlag(EFlowStep.IPCores))
166  project.SkipIPCoreSynthesis = true;
167  project.TopLevelComponent = TopLevelComponent.Descriptor;
168  CreateUCF(project);
169 
170  VHDLGenerator codeGen = new VHDLGenerator();
171  SynthesisEngine.Create(DesignContext.Instance, project).Synthesize(GetComponentSet(), codeGen);
172  project.Save();
173  if (step.HasFlag(EFlowStep.XST) ||
174  step.HasFlag(EFlowStep.NGDBuild) ||
175  step.HasFlag(EFlowStep.Map) ||
176  step.HasFlag(EFlowStep.PAR) ||
177  step.HasFlag(EFlowStep.TRCE))
178  {
179  var flow = project.ConfigureFlow(TopLevelComponent);
180  flow.Start(step);
181  }
182  return project;
