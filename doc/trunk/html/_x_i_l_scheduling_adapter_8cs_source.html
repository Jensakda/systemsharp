215  private ScheduleDependency<XIL3Instr> ToScheduleDependency(XIL3Instr instr, InstructionDependency dep)
216  {
217  var predInstr = _func.Instructions[dep.PredIndex];
218  var odep = dep as OrderDependency;
219  var tdep = dep as TimeDependency;
220  if (odep != null)
221  {
222  switch (odep.Kind)
223  {
224  case OrderDependency.EKind.BeginAfter:
225  return new ScheduleDependency<XIL3Instr>(predInstr, Latency[predInstr], int.MaxValue);
226 
227  case OrderDependency.EKind.CompleteAfter:
228  return new ScheduleDependency<XIL3Instr>(predInstr, Latency[predInstr] - Latency[instr], int.MaxValue);
229 
230  default:
231  throw new NotImplementedException();
232  }
233  }
234  else if (tdep != null)
235  {
236  return new ScheduleDependency<XIL3Instr>(predInstr, tdep.MinDelay, tdep.MaxDelay);
237  }
238  else
239  {
240  throw new NotImplementedException();

244  private void Init()
245  {
246  int numInstrs = _func.Instructions.Length;
247  _predsBack = new ScheduleDependency<XIL3Instr>[numInstrs][];
248  _succsBack = new ScheduleDependency<XIL3Instr>[numInstrs][];
249  var tmp = new List<ScheduleDependency<XIL3Instr>>[numInstrs];
250  var rslot2instr = new Dictionary<int, XIL3Instr>();
251  foreach (XIL3Instr i in _func.Instructions)
252  {
253  foreach (int rslot in i.ResultSlots)
254  {
255  rslot2instr[rslot] = i;
256  }
257  var bb = _cfg.GetBasicBlockContaining(i.Index);
258  _predsBack[i.Index] = i.Preds
259  .Select(p => ToScheduleDependency(i, p))
260  .Concat(i.OperandSlots.Select(s =>
261  new ScheduleDependency<XIL3Instr>(
262  rslot2instr[s],
263  Latency[rslot2instr[s]],
264  int.MaxValue)))
265  .Distinct()
266  .Where(_ => bb.Contains(_.Task)) // only admit predecessors from same basic block
267  .ToArray();
268  tmp[i.Index] = new List<ScheduleDependency<XIL3Instr>>();
269  }
270  foreach (XIL3Instr i in _func.Instructions)
271  {
272  foreach (var sdep in _predsBack[i.Index])
273  {
274  tmp[sdep.Task.Index].Add(new ScheduleDependency<XIL3Instr>(i, sdep.MinDelay, sdep.MaxDelay));
275  }
276  }
277  foreach (XIL3Instr i in _func.Instructions)
278  {
279  _succsBack[i.Index] = tmp[i.Index].ToArray();
280  }
281  _preds = PropMaps.CreateForArray<XIL3Instr, ScheduleDependency<XIL3Instr>[]>(_predsBack, i => i.Index, EAccess.ReadOnly);
282  _succs = PropMaps.CreateForArray<XIL3Instr, ScheduleDependency<XIL3Instr>[]>(_succsBack, i => i.Index, EAccess.ReadOnly);

422  public bool TryPin(XIL3Instr task, long cstep, out long preHint, out long postHint)
423  {
424  object iclass = Classify(task);
425  var mapping = GetMapping(task);
426  int max;
427  if (mapping.ResourceKind != EMappingKind.LightweightResource &&
428  _maxFUs.TryGetValue(iclass, out max))
429  {
430  int ii = GetMapping(task).InitiationInterval;
431  bool found;
432  long preStep = cstep, postStep = cstep;
433  do
434  {
435  found = true;
436  for (long i = preStep; i < preStep + ii; i++)
437  {
438  int alloc;
439  _resTable.TryGetValue(Tuple.Create(iclass, i), out alloc);
440  if (alloc == max)
441  {
442  found = false;
443  --preStep;
444  break;
445  }
446  }
447  for (long i = postStep; i < postStep + ii; i++)
448  {
449  int alloc;
450  _resTable.TryGetValue(Tuple.Create(iclass, i), out alloc);
451  if (alloc == max)
452  {
453  found = false;
454  ++postStep;
455  break;
456  }
457  }
458  } while (!found);
459  preHint = preStep;
460  postHint = postStep;
461  if (preStep == postStep)
462  {
463  for (long i = cstep; i < cstep + ii; i++)
464  {
465  var key = Tuple.Create(iclass, i);
466  int alloc;
467  _resTable.TryGetValue(key, out alloc);
468  ++alloc;
469  _resTable[key] = alloc;
470  }
471  return true;
472  }
473  else
474  {
475  return false;
476  }
477  }
478  else
479  {
480  preHint = cstep;
481  postHint = cstep;
482  _cstepBack[task.Index] = cstep;
483  return true;
484  }
485  }

506  public FlowMatrix Allocate(IDatapathBuilder dpb)
507  {
508  // Step 1: create intermediate storage for each instruction output slot
509  var tempSignals = new SLVSignal[_func.SlotTypes.Length];
510  for (int i = 0; i < _func.SlotTypes.Length; i++)
511  {
512  int width = TypeLowering.Instance.GetWireWidth(_func.SlotTypes[i]);
513  var initial = StdLogicVector.Us(width);
514  tempSignals[i] = (SLVSignal)Signals.CreateInstance(initial);
515  tempSignals[i].Descriptor.TagTemporary(i);
516  }
517 
518  // Step 2: Allocate and establish unit bindings
519  Action<IXILMapping> onAlloc = map => dpb.AddFU(map.TASite.Host);
520  _alloc.OnFUAllocation += onAlloc;
521  var mappings = new Dictionary<int, IXILMapping>();
522  foreach (var xil3i in _func.Instructions)
523  {
524  TypeDescriptor[] otypes = xil3i.OperandSlots.Select(i => _func.SlotTypes[i]).ToArray();
525  TypeDescriptor[] rtypes = xil3i.ResultSlots.Select(i => _func.SlotTypes[i]).ToArray();
526  long cstep = CStep[xil3i];
527  var mapping = _alloc.TryBind(xil3i, cstep, otypes, rtypes);
528  if (mapping == null)
529  throw new XILSchedulingFailedException("Realization failed: " + xil3i.Command);
530  mappings[xil3i.Index] = mapping;
531  mapping.TASite.Establish(dpb.FUBinder);
532  Allocator.Policy.TellMapping(xil3i, cstep, mapping);
533  }
534  _alloc.OnFUAllocation -= onAlloc;
535 
536  // Step 3: Assemble flow matrix
537  var result = new FlowMatrix();
538  foreach (var xil3i in _func.Instructions)
539  {
540  long cstep = CStep[xil3i];
541  var mapping = mappings[xil3i.Index];
542  result.AddNeutral(mapping.TASite.DoNothing());
543  var impl = mapping.Realize(
544  xil3i.OperandSlots.Select(s => tempSignals[s].AsSignalSource<StdLogicVector>()).ToArray(),
545  xil3i.ResultSlots.Select(s => tempSignals[s].AsCombSink()).ToArray());
546  if (impl.Count() - Latency[xil3i] > 1)
547  throw new XILSchedulingFailedException("Mapping length exceeds scheduled latency");
548  result.Add((int)cstep, impl);
549  result.AppendComment((int)cstep, ToComment(xil3i));
550  _mappingCache[xil3i.Index] = mapping;
551  }
552 
553  return result;
554  }
