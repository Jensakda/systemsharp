565  public void Encode()
566  {
567  // This is a simple, inefficient greedy heuristic which essentially finds maximal cliques
568  // within the graph of mutually exclusive signal targets.
569  var mutexSet = new HashSet<Tuple<SignalRef, SignalRef>>();
570  foreach (var pflow in _flowList)
571  {
572  var flows = pflow.Flows.ToArray();
573  for (int i = 0; i < flows.Length; i++)
574  {
575  for (int j = 0; j < flows.Length; j++)
576  {
577  var vf0 = (ValueFlow)flows[i];
578  var vf1 = (ValueFlow)flows[j];
579  if (!FlowMatrix.IsDontCareFlow(vf0) &&
580  !FlowMatrix.IsDontCareFlow(vf1) &&
581  !vf0.Value.Equals(vf1.Value))
582  {
583  mutexSet.Add(Tuple.Create(vf0.Target, vf1.Target));
584  }
585  }
586  }
587  }
588  var order = _widthMap.OrderByDescending(kvp => kvp.Value).Select(kvp => kvp.Key);
589  var list = new LinkedList<SignalRef>(order);
590  int curOffset = 0;
591  while (list.First != null)
592  {
593  var lln = list.First;
594  var curGroup = new HashSet<SignalRef>();
595  curGroup.Add(lln.Value);
596  int width = _widthMap[lln.Value];
597  _offsetMap[lln.Value] = curOffset;
598  var next = lln.Next;
599  list.Remove(lln);
600  lln = next;
601  while (lln != null)
602  {
603  bool found = true;
604  foreach (var elem in curGroup)
605  {
606  if (mutexSet.Contains(Tuple.Create(elem, lln.Value)))
607  {
608  found = false;
609  break;
610  }
611  }
612  if (found)
613  {
614  curGroup.Add(lln.Value);
615  _offsetMap[lln.Value] = curOffset;
616  next = lln.Next;
617  list.Remove(lln);
618  lln = next;
619  }
620  else
621  {
622  lln = lln.Next;
623  }
624  }
625  curOffset += width;
626  }
627  ValueWordWidth = curOffset;
628  }

718  }
719  }

722  var nextGen = (EncodedFlow[])curGen.Clone();
723  var mergedIndices = new HashSet<int>();
724  var mergedLowIndices = new SortedSet<int>();
725  var mergedHiIndices = new HashSet<int>();

736  if (selWidth > maxSelWidth)

744  mergedIndices.Add(tup.Item2);
745  mergedLowIndices.Add(tup.Item1);
746  mergedHiIndices.Add(tup.Item2);
747  indices.Remove(tup.Item2);
748

751  Debug.Assert(nextGen[tup.Item2].Targets.All(t => mergedFlow.Targets.Contains(t)));
752  nextGen[tup.Item1] = mergedFlow;
753  mergedAny = true;
754  }
755  nextCandidates.Clear();
756  curGen = nextGen;
757  mergeCandidates.Clear();
758  mergeCandidates.AddRange(nextCandidates);
759  }
760  while (mergedAny);
761 
762  _strings = indices.Select(i => new MicroString(curGen[i], _vcf)).ToArray();
763

765  var coveredTargets = _strings.SelectMany(s => s.Targets);
766  var allTargets = encFlows.SelectMany(f => f.Targets);
767  var isect0 = coveredTargets.Except(allTargets);
768  var isect1 = allTargets.Except(coveredTargets);
769  Debug.Assert(!isect0.Any());
770  Debug.Assert(!isect1.Any());

774  int order = 0;
775  foreach (var ms in _strings)
776  {
777  ms.SelOffset = offset;
778  ms.Order = order;
779  offset += ms.SelWidth;
780  order++;
781  }
782  CWWidth = offset;
783

785  var runTime = stopTime - startTime;
786  sb.AppendFormat(" Compressed CW: {0} MUX bits + {1} value bits",
787  offset - _vcf.ValueWordWidth, _vcf.ValueWordWidth);
788  sb.AppendLine();
789  sb.AppendFormat(" Maximum LUT inputs: {0}", _strings.Max(s => s.SelWidth));

791  sb.AppendLine();

796  foreach (var grp in histo)

799  sb.AppendLine();
