144  private void Process()
145  {
146  StdLogic ib1 = X.Cur[FracWidth + 1];
147  StdLogic ib0 = X.Cur[FracWidth];
148  StdLogicVector rv = X.Cur[FracWidth - 1, 0];
149  bool ibf1 = ib1 == '1';
150  bool ibf0 = ib0 == '1';
151  bool rvz = rv == _zeroes;
152  bool flag = ibf0 && (ibf1 || !rvz);
153  if (flag)
154  {
155  R.Next = _padOnes.Concat(ib0.Concat(rv));
156  }
157  else
158  {
159  R.Next = _padZeroes.Concat(ib0.Concat(rv));
160  }
161  }
