97  action = action.Par(_host.C_in.Dual.Stick<StdLogic>('0'));

472  [DoNotAnalyze]
473  private void Processing()
474  {
475  if (Latency == 0 || (CLK.RisingEdge() && (!HasCE || CE.Cur == '1')))
476  {
477  int width = Math.Max(Awidth, Bwidth);
478  if (AddMode == EAddMode.Add ||
479  (AddMode == EAddMode.AddSubtract && ADD.Cur == '1'))
480  {
481  _result.Next =

487  _result.Next =

819  Awidth = fmta.TotalWidth,
